   11 genHeightsSplit ::
   12     (Int, Int) -> (Int, Int) -> Int -> Int -> GLfloat -> GLfloat -> IO [[[GLfloat]]]
   13 genHeightsSplit (width, height) (startx, starty) splits octaves wavelength intensity
   14     | max startx starty < width = do
   15         cur <- simplexNoiseSection (width, height) (startx, starty) octaves wavelength intensity
   16         rest <- genHeightsSplit (width, height)
   17                 (startx + width, starty + width) splits octaves wavelength intensity
   18         return $ cur : rest
   19     | otherwise = return []
   20 -}

   22 genSimplexModel :: FilePath -> FilePath ->
   23     GLfloat ->          -- ^ Width
   24     GLfloat ->          -- ^ Spacing
   25     Int ->              -- ^ Octaves
   26     GLfloat ->          -- ^ Wavelength
   27     GLfloat ->          -- ^ Waveheight / intensity
   28     Maybe FilePath ->   -- ^ The texture (Maybe)
   29     IO Model
   30 genSimplexModel vert frag w spacing octaves wavelength intensity texture = do
   31     heights <- simplexNoise (floor w) spacing octaves wavelength intensity
   32     let hCoords = heightsToCoords heights 0 spacing
   33         flat = createFlat spacing w
   34         vertices = applyHeights flat hCoords
   35         normals = calculateNormals vertices
   36         lengthVertices = length vertices
   37 
   38     if isJust texture
   39         then do
   40             loadedModel <- createModel vert frag ["position", "normal", "color", "texCoord", "textureId"]
   41                     [vertices, normals, take (lengthVertices * 3) (cycle [0, 1, 0]),
   42                     take (lengthVertices * 3) $ cycle [0, 0, 1, 0, 0, 1],
   43                     replicate lengthVertices 0]
   44                     [3, 3, 3, 2, 1]
   45                     (fromIntegral $ length vertices `div` 3)
   46             textureData <- juicyLoadTexture $ fromJust texture
   47             return $ loadedModel{modelTextures = [(textureData, 1)]}
   48     else
   49         createModel vert frag ["position", "normal", "color", "texCoord", "textureId"]
   50                     [vertices, normals, take (lengthVertices * 3) (cycle [0, 1, 0]),
   51                     replicate (lengthVertices * 3) 0,
   52                     replicate lengthVertices (-1)]
   53                     [3, 3, 3, 2, 1]
   54                     (fromIntegral $ length vertices `div` 3)

   89 createFlat :: GLfloat -> GLfloat -> [GLfloat]
   90 createFlat spacing width =
   91     createFlatRaw spacing width 0
   92 
   93     where
   94         createFlatRaw s w x =
   95             createStrip s w x 0 ++
   96                 if x < w 
   97                     then createFlatRaw s w (x+spacing)
   98                 else []
   99 
  100 createStrip :: GLfloat -> GLfloat -> GLfloat -> GLfloat -> [GLfloat]
  101 createStrip spacing len x i
  102     | i < len =
  103         [x, 0, i,
  104          x, 0, i+spacing,
  105          x+spacing, 0, i,
  106          x+spacing, 0, i,
  107          x, 0, i+spacing,
  108          x+spacing, 0, i+spacing] ++ createStrip spacing len x (i+spacing)
  109     | otherwise = []

  111 applyHeights :: [GLfloat] -> [GLfloat] -> [GLfloat]
  112 applyHeights (x:y:z:coords) ref =
  113     let foundY = findY x z ref
  114     in x : (if foundY == -3.1415
  115         then y
  116     else foundY) : z : applyHeights coords ref
  117 applyHeights _ _ = []
  118 
  119 findY :: GLfloat -> GLfloat -> [GLfloat] -> GLfloat
  120 findY xf zf (xl:yl:zl:heights)
  121     | xf == xl && zf == zl = yl
  122     | otherwise = findY xf zf heights
  123 findY _ _ _ = -3.1415
  124 
  125 calculateNormals :: [GLfloat] -> [GLfloat]
  126 calculateNormals (x1:y1:z1:x2:y2:z2:x3:y3:z3:rest) =
  127     let (ux, uy, uz) = (x2 - x1, y2 - y1, z2 - z1)
  128         (vx, vy, vz) = (x3 - x1, y3 - y1, z3 - z1)
  129         nx = (uy * vz) - (uz * vy)
  130         ny = (uz * vx) - (ux * vz)
  131         nz = (ux * vy) - (uy * vx)
  132     in repeatList 3 [nx, if ny > 0 then ny else -ny, nz] ++
  133             calculateNormals rest
  134 calculateNormals _ = []
  135 
  136 heightsToCoords :: [[GLfloat]] -> GLfloat -> GLfloat -> [GLfloat]
  137 heightsToCoords (yi:restYs) z spacing =
  138     let completeLine (y:ys) x =
  139             [x, y, z] ++ completeLine ys (x + spacing)
  140         completeLine [] _ = []
  141     in completeLine yi 0 ++ heightsToCoords restYs (z + spacing) spacing
  142 heightsToCoords [] _ _ = []
  143 
  144 repeatList :: Int -> [a] -> [a]
  145 repeatList i arr = concat $ replicate i arr
