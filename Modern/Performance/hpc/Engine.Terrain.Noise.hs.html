   25 dot3 :: (Double, Double, Double) -> Double -> Double -> Double -> Double
   26 dot3 (a,b,c) x y z = a * x + b * y + c * z

   28 {-# INLINE fastFloor #-}
   29 fastFloor :: Double -> Int
   30 fastFloor x = truncate (if x > 0 then x else x - 1)

   32 -- | Generate a random permutation for use in the noise functions
   33 perm :: Int -> Permutation
   34 perm seed = V.fromList . concat . replicate 2 . shuffle' [0..255] 256 $
   35                 mkStdGen seed

   37 -- | Generate 3D noise between -0.5 and 0.5
   38 noise3D :: Permutation -> Double -> Double -> Double -> Double
   39 noise3D p x y z =
   40     16 * (n gi0 (x0,y0,z0) + n gi1 xyz1 + n gi2 xyz2 + n gi3 xyz3)
   41     where
   42     (i,j,k) = (s x, s y, s z)
   43     s a = fastFloor (a + (x + y + z) / 3)
   44     (x0, y0, z0) = (x - int i + t, y - int j + t, z - int k + t)
   45     t = int (i + j + k) * g3
   46     (i1, j1, k1, i2, j2, k2)
   47         | x0 >= y0 =
   48             if y0 >= z0
   49                 then (1,0,0,1,1,0)
   50             else if x0 >= z0
   51                 then (1,0,0,1,0,1)
   52             else (0,0,1,1,0,1)
   53         | x0 >= y0 =
   54                 if y0 >= z0
   55                     then (1,0,0,1,1,0)
   56                 else if x0 >= z0
   57                     then (1,0,0,1,0,1)
   58                 else (0,0,1,1,0,1)
   59         | y0 <  z0 = (0,0,1,0,1,1)
   60         | x0 <  z0 = (0,1,0,0,1,1)
   61         | otherwise = (0,1,0,1,1,0)
   62     xyz1 = (x0 - int i1 +   g3, y0 - int j1 +   g3, z0 - int k1 +   g3)
   63     xyz2 = (x0 - int i2 + 2*g3, y0 - int j2 + 2*g3, z0 - int k2 + 2*g3)
   64     xyz3 = (x0 - 1 + 3*g3, y0 - 1 + 3*g3, z0 - 1 + 3*g3)
   65     (ii,jj,kk) = (i .&. 255, j .&. 255, k .&. 255)
   66     gi0 = rem (V.unsafeIndex p 
   67         (ii + V.unsafeIndex p (jj + V.unsafeIndex p kk))) 12
   68     gi1 = rem (V.unsafeIndex p 
   69         (ii + i1 + V.unsafeIndex p (jj + j1 + V.unsafeIndex p (kk + k1)))) 12
   70     gi2 = rem (V.unsafeIndex p
   71         (ii + i2 + V.unsafeIndex p (jj + j2 + V.unsafeIndex p (kk + k2)))) 12
   72     gi3 = rem (V.unsafeIndex p
   73         (ii + 1  + V.unsafeIndex p (jj + 1  + V.unsafeIndex p (kk + 1)))) 12
   74     {-# INLINE n #-}
   75     n gi (x',y',z') = (\a -> if a < 0 then 0 else
   76         a*a*a*a*dot3 (V.unsafeIndex grad3 gi) x' y' z') $
   77             0.6 - x'*x' - y'*y' - z'*z'
   78 
   79 harmonic :: Int -> (Double -> Double) -> Double
   80 harmonic octaves noise = f octaves / (2 - 1 / int (2 ^ (octaves - 1)))
   81     where
   82     f 0 = 0
   83     f o = let r = 2 ^^ (o - 1) in noise r / r + f (o - 1)
   84 
   85 -- | 3D simplex noise
   86 --   args - permutation number_of_octaves wavelength x y z
   87 simplex3D :: Permutation -> Int -> Double -> Double -> Double -> Double -> Double
   88 simplex3D p octaves l x y z = harmonic octaves
   89     (\f -> noise3D p (x * f / l) (y * f / l) (z * f / l))
   90 
   91 -- | Generate a 2D list of the height returned by simplex
   92 --   for each coordinate.
   93 simplexNoise :: Int -> GLfloat -> Int -> GLfloat -> GLfloat -> IO [[GLfloat]]
   94 simplexNoise width spacing octaves wavelength intensity = do
   95     seed <- randomRIO (0, 100)
   96     let raw = map
   97             (\(x, y) -> intensity * realToFrac
   98                 (simplex3D (perm seed) octaves (realToFrac wavelength) x y 0))
   99             [(realToFrac x, realToFrac y) |
  100                 x <- [0, spacing .. (fromIntegral $ width-1)],
  101                 y <- [0, spacing .. (fromIntegral $ width-1)]]
  102     return $ splitInterval raw width
  103 
  104 simplexNoiseSection :: (Int, Int) -> (Int, Int) -> Int -> GLfloat -> GLfloat -> IO [[GLfloat]]
  105 simplexNoiseSection (width, height) (startx, starty) octaves wavelength intensity = do
  106     seed <- randomRIO (0, 100)
  107     let raw = map
  108             (\(x, y) -> intensity * realToFrac
  109                 (simplex3D (perm seed) octaves (realToFrac wavelength) x y 0))
  110             [(realToFrac x, realToFrac y) |
  111                 x <- [startx .. (width-1)],
  112                 y <- [starty .. (height-1)]]
  113     return $ splitInterval raw width
  114 
  115 splitInterval :: [a] -> Int -> [[a]]
  116 splitInterval xs i
  117     | not $ null xs =
  118         let (ret, rest) = splitAt i xs
  119         in ret : splitInterval rest i
  120     | otherwise = []
