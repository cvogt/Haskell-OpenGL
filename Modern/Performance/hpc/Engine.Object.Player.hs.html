   26 pUpdate :: Game t (World t)
   27 pUpdate = do
   28     w <- get
   29     if not $ statePaused $ worldState w
   30         then
   31         let p = worldPlayer w
   32             state = worldState w
   33             origSpeed = playerSpeed p
   34             speed = origSpeed * stateDelta state
   35             newP = p{playerSpeed = speed}
   36             -- Do actual update
   37             modifiedW = playerKeyUpdateSafe w{worldPlayer = playerMouseUpdate newP}
   38             modifiedP = worldPlayer modifiedW
   39             gravityP = applyGravityVelocity w modifiedP
   40 
   41             resolvedP = resolveVelocity w gravityP
   42 
   43         in return modifiedW{worldPlayer =
   44             resolvedP{playerSpeed = origSpeed}}
   45     else
   46         let p = worldPlayer w
   47             -- Do mouse update.
   48             playerMouseUpdated = playerMouseUpdate p
   49             -- Do key update.
   50             modifiedW = playerKeyUpdateSafe w{worldPlayer = playerMouseUpdated}
   51 
   52         in return modifiedW{worldPlayer = p}

  102         let p = worldPlayer w
  103             curVel@(Vec3 _ vy _) = playerVelocity p
  104         in
  105             if abs vy < 0.01
  106                 then w{worldPlayer =
  107                     p{playerVelocity =
  108                         curVel + Vec3 0 0.2 0}}

  123 -- | Calculate delta movement from Player and eaw input movement.
  124 calculateLookMovement :: GameObject t -> Vec3 GLfloat -> Vec3 GLfloat
  125 calculateLookMovement p@(Player{}) (Vec3 idx idy idz) =
  126     let Vec3 _ rry _ = playerRotation p
  127         dx = realToFrac idx
  128         dz = realToFrac idz
  129         
  130         ry = realToFrac rry :: Float
  131 
  132         mx = dx * sinDeg (ry - 90) + dz * sinDeg ry
  133         my = idy
  134         mz = dx * cosDeg (ry - 90) + dz * cosDeg ry
  135     in Vec3 (realToFrac mx) my (realToFrac mz)
  136 calculateLookMovement _ _ =
  137     error "Player.calculateLookMovement can only be used on Players."
  138 
  139 -- | Takes a Player and a Vec3 of movement
  140 --   and moves player locally based on rotation.
  141 moveFromLook :: GameObject t -> Vec3 GLfloat -> GameObject t
  142 moveFromLook player@(Player{}) givenVec =
  143     let movement = calculateLookMovement player givenVec
  144     in moveObject player movement
  145 moveFromLook _ _ =
  146     error "Player.moveFromLook can only be used on Players."

  150 moveFromLookSlide :: World t -> GameObject t -> Vec3 GLfloat -> GameObject t
  151 moveFromLookSlide world player@(Player{}) idVec =
  152     let movement = calculateLookMovement player idVec

  154     in moveWithStep world player movement

  158 -- | Calculate new velocity from current object
  159 --   and raw movement.
  160 setVelocityFromLook :: GameObject t -> Vec3 GLfloat -> GameObject t
  161 setVelocityFromLook player@(Player{}) idVec =
  162     let movement = calculateLookMovement player idVec
  163     in player{playerVelocity =
  164             playerVelocity player + movement}

  167 -- | Move Player from raw delta movement, simulating
  168 --   the Player stepping up small steps.
  169 moveWithStep :: World t -> GameObject t -> Vec3 GLfloat -> GameObject t
  170 moveWithStep world player@(Player{}) movement@(Vec3 mx _ mz) = do
  171     let (moved, mabInt) = moveObjectSlideAllIntersecters
  172                 world player movement
  173     if isJust mabInt
  174         then
  175             let abMaxYs = map (\(AABB _ (Vec3 _ ymax _)) -> ymax) $ fromJust mabInt
  176                 abMaxY = maximum abMaxYs
  177                 (Just (AABB (Vec3 _ pMiny _) _)) = calculateNewWholeAABB moved
  178                 yStep = abMaxY - pMiny
  179             in if abs yStep < 1
  180                     then
  181                     let moved2 =
  182                             moveObjectSlide world moved $
  183                                     Vec3 mx (yStep+1e-2) mz
  184                         Vec3 velX _ velZ = playerVelocity moved2
  185                         
  186                     in moved2{playerVelocity = Vec3 velX 0 velZ}
  187                 else
  188                     let Vec3 velX _ velZ = playerVelocity moved
  189                     in moved{playerVelocity = Vec3 velX 0 velZ}
  190     else moved
  191 moveWithStep _ _ _ =
  192     error "Player.moveWithStep can only be used on Players."

  194 -- | Update rotation from mouse input.
  195 playerMouseUpdate :: GameObject t -> GameObject t
  196 playerMouseUpdate player =
  197     let Vec2 rawdx rawdy = inputMouseDelta $ playerInput player
  198         Vec2 lastX lastY = inputLastMousePos $ playerInput player
  199         -- TODO: adjust multipliers
  200         (dxx, dy) = (rawdx*0.1, rawdy*0.1)
  201 
  202         Vec3 rx ry rz = playerRotation player
  203 
  204         dx = -dxx
  205 
  206         curPos = Vec2 (lastX + rawdx) (lastY + rawdy)
  207 
  208         -- Basic calculation of degrees, 0 is minimum,
  209         -- 360 is maximum.
  210         newRy   
  211             | ry + dx >= 360 = ry + dx - 360
  212             | dx + ry < 0    = 360 - ry + dx
  213             | otherwise      = ry + dx
  214 
  215         -- Lowest angle player can look
  216         maxLookDown = -90
  217         -- Highest angle player can look
  218         maxLookUp = 90
  219 
  220         -- Basic calculation for x axis (looking up and down).
  221         -- Make sure that look direction stays between maxLookDown
  222         -- and maxLookUp.
  223         newRx
  224             -- If rotation is in bounds, allow rotation.
  225             | rx - dy >= maxLookDown && rx - dy <= maxLookUp      = rx - dy
  226             -- If player is trying to look down too far, set rotation to maxLookDown.
  227             | rx - dy < maxLookDown                             = maxLookDown
  228             -- If player is trying to look up too far, set rotation to maxLookUp.
  229             | rx - dy > maxLookUp                               = maxLookUp
  230             -- I don't think this will ever happen.
  231             | otherwise                                         = rx
  232 
  233         -- Update inputLastMousePos
  234         newInput = (playerInput player){inputLastMousePos = curPos}
  235 
  236         -- Return given player with modified rotation.
  237         newRot = Vec3 newRx newRy rz
  238     in player{playerRotation = newRot, playerInput = newInput}

  240 -- | Player update with collision detection.
  241 playerKeyUpdateSafe :: World t -> World t
  242 playerKeyUpdateSafe world =
  243     let startPlayer = worldPlayer world
  244         retW = playerKeyUpdateTailSafe world (worldPlayer world)
  245         retP = (worldPlayer retW){playerInput = playerInput startPlayer}
  246     in retW{worldPlayer = retP}
  247 
  248 -- | Returns Player after safely applying all input functions.
  249 --   UNSAFE! Returns given player with an empty inputKeys!
  250 --   Use playerKeyUpdateSafe instead.
  251 playerKeyUpdateTailSafe :: World t -> GameObject t -> World t
  252 playerKeyUpdateTailSafe w
  253     (Player _ _ _ _ _ (Input ((_, desired, found, func):xs) mouse lm ms)) =
  254     -- If the recorded keystate matches desired keystate,
  255     -- apply corresponding function to player.
  256     -- Assumes KeyState'Repeating = KeyState'Repeating || KeyState'Pressed
  257     let newWorld
  258             | desired == GLFW.KeyState'Repeating =
  259                 if found == desired ||
  260                         found == GLFW.KeyState'Pressed
  261                     then func w
  262                 else w
  263             | desired == found =
  264                 func w
  265             | otherwise = w
  266     -- Give modified player to the function again, to recursively
  267     -- apply each key update.
  268         newPlayer = worldPlayer newWorld
  269         retp = newPlayer{playerInput = Input xs mouse lm ms}
  270     in playerKeyUpdateTailSafe newWorld{worldPlayer = retp} retp
  271 playerKeyUpdateTailSafe w (Player _ _ _ _ _ (Input [] _ _ _)) = w
  272 playerKeyUpdateTailSafe _ _ =
  273     error $ "Player.playerKeyUpdateTailSafe is meant"
  274         ++ " to be used only on Players."

  276 -- | Change Player's velocity to simulate gravity.
  277 applyGravityVelocity :: World t -> GameObject t -> GameObject t
  278 applyGravityVelocity world p@(Player{}) =
  279     let Vec3 pvx pvy pvz = playerVelocity p
  280         atRest
  281             | pvy < 0 =
  282                 let m = moveObjectSafe world p (Vec3 0 pvy 0)
  283                 in getPos m == getPos p
  284             | pvy == 0 =
  285                 let m = moveObjectSafe world p (Vec3 0 (-0.05) 0)
  286                 in getPos m == getPos p
  287             | otherwise = False
  288     in if atRest
  289         then p{playerVelocity = Vec3 pvx 0 pvz}
  290     else
  291         let Vec3 vx vy vz = playerVelocity p
  292             newY = max (vy - getWorldDelta world * 0.5) (-1)
  293         in p{playerVelocity =
  294             Vec3 vx newY vz}
  295 applyGravityVelocity _ _ =
  296     error "Player.applyGravityVelocity can only be used on Players."
  297 
  298 -- | Translate velocity into a new position.
  299 resolveVelocity :: World t -> GameObject t -> GameObject t
  300 resolveVelocity world p@(Player{}) =
  301     let pVel@(Vec3 _ yv _) = playerVelocity p
  302         movedPlayer = moveWithStep world p pVel
  303     in movedPlayer{playerVelocity =
  304                 Vec3 0 yv 0}
  305 resolveVelocity _ _ =
  306     error $ "Player.resolveVelocity can only be used"
  307             ++ " on Players"
