   16 writeDataToFile :: FilePath -> [[GLfloat]] -> [FilePath] -> IO ()
   17 writeDataToFile datFile [verts, coords, norms, colors, tids] =
   18     writeVntToDat datFile (Vec4 verts norms coords tids) colors
   19 
   20 writeVntToDat :: FilePath -> Vec4 [GLfloat] -> [GLfloat] -> [FilePath] -> IO ()
   21 writeVntToDat datFile (Vec4 verts norms texCoords texIds) colors images =
   22     let vertStr = listToString verts
   23         normStr = listToString norms
   24         texcStr = listToString texCoords
   25         texiStr = listToString texIds
   26         colsStr = listToString colors
   27         imgsStr = listStringsToString images
   28 
   29     in do
   30         handle <- openFile datFile WriteMode
   31         hPutStr handle "v "
   32         hPutStrLn handle vertStr
   33 
   34         hPutStr handle "n "
   35         hPutStrLn handle normStr
   36 
   37         hPutStr handle "tc "
   38         hPutStrLn handle texcStr
   39 
   40         hPutStr handle "ti "
   41         hPutStrLn handle texiStr
   42 
   43         hPutStr handle "img "
   44         hPutStrLn handle imgsStr
   45 
   46         hPutStr handle "rgb "
   47         hPutStrLn handle colsStr
   48 
   49 listToString :: Show a => [a] -> String
   50 listToString (x:y:xs) =
   51     show x ++ "," ++ listToString (y:xs)
   52 listToString (x:[]) =
   53     show x
   54 
   55 listStringsToString :: [String] -> String
   56 listStringsToString (x:y:xs) =
   57     x ++ "," ++ listStringsToString (y:xs)
   58 listStringsToString (x:[]) =
   59     x

   63 loadData :: FilePath -> IO ([[GLfloat]], [FilePath])
   64 loadData datFile = do
   65     fileContents <- openFile datFile ReadMode >>= B.hGetContents
   66     let fileLines = B.lines fileContents
   67         verts = loadVerts fileLines
   68         norms = loadNormals fileLines
   69         colors = loadColors fileLines
   70         texCoords = loadTexCoords fileLines
   71         texIds = loadTexIds fileLines
   72         textures = loadTextures fileLines
   73     return ([verts, texCoords, norms, colors, texIds], map B.unpack textures)
   74 
   75 loadVerts :: [B.ByteString] -> [GLfloat]
   76 loadVerts =
   77     readDataLine . takeFirst (B.isPrefixOf "v ")
   78 
   79 loadNormals :: [B.ByteString] -> [GLfloat]
   80 loadNormals =
   81     readDataLine . takeFirst (B.isPrefixOf "n ")
   82 
   83 loadColors :: [B.ByteString] -> [GLfloat]
   84 loadColors =
   85     readDataLine . takeFirst (B.isPrefixOf "rgb ")
   86 
   87 loadTexCoords :: [B.ByteString] -> [GLfloat]
   88 loadTexCoords =
   89     readDataLine . takeFirst (B.isPrefixOf "tc ")
   90 
   91 loadTexIds :: [B.ByteString] -> [GLfloat]
   92 loadTexIds =
   93     readDataLine . takeFirst (B.isPrefixOf "ti ")
   94 
   95 loadTextures :: [B.ByteString] -> [B.ByteString]
   96 loadTextures =
   97     parseDataLine . takeFirst (B.isPrefixOf "img ")
   98 
   99 readDataLine :: B.ByteString -> [GLfloat]
  100 readDataLine = map parseBsFloat . parseDataLine
  101 
  102 parseDataLine :: B.ByteString -> [B.ByteString]
  103 parseDataLine = B.split ',' . last . B.words
  104 
  105 parseBsFloat :: B.ByteString -> GLfloat
  106 parseBsFloat = fst . parseBsFloat'
  107 
  108 parseBsFloat' :: B.ByteString -> (GLfloat, B.ByteString)
  109 parseBsFloat' bs
  110     | isJust ('e' `B.elemIndex` bs) =
  111         let [baseBs, expBs] = B.splitWith (=='e') bs
  112             (base, _) = parseBsFloat' baseBs
  113             (expon, rest) = parseBsFloat' expBs
  114         in (base * (10.0 ** expon), rest)
  115 
  116     | isJust ('.' `B.elemIndex` bs) =
  117         let negative = B.head bs == '-'
  118             Just (whole, decimalBS) = B.readInt $
  119                 if negative
  120                     then B.tail bs
  121                 else bs
  122             Just (decimal, rest) = B.readInt $ B.tail decimalBS
  123         in if negative
  124                 then (negate $ fromIntegral whole +
  125                     (fromIntegral decimal / 10 ^ (B.length decimalBS - 1)), rest)
  126             else (fromIntegral whole +
  127                 (fromIntegral decimal / 10 ^ (B.length decimalBS - 1)), rest)
  128 
  129     | otherwise =
  130         let Just (val, rest) = B.readInt bs
  131         in (fromIntegral val, rest)
  132 {-# INLINE parseBsFloat' #-}

  134 takeFirst :: (a -> Bool) -> [a] -> a
  135 takeFirst func (x:xs)
  136     | func x = x
  137     | otherwise = takeFirst func xs
  138 takeFirst _ _ = error "DatLoader.takeFirst: pattern not matched."
  139 {-# INLINE takeFirst #-}
