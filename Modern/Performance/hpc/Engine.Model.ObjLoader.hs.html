   83 -- | Parse an .obj file and return a Vec3 [GLfloat]
   84 --   containing the vertices, texture coordinates,
   85 --   and normals, in that order.
   86 loadObjFile :: FilePath -> IO (Vec3 [GLfloat])
   87 loadObjFile file = do
   88     handle <- openFile file ReadMode
   89     objData <- liftM loadObj $! hGetContents handle
   90     hClose handle
   91     return objData
   92 
   93 -- | Parse a string containing the specification
   94 --   for an .obj model and return a Vec3 [GLfloat]
   95 --   containing the vertices, texture coordinates,
   96 --   and normals, in that order.
   97 loadObj :: String -> Vec3 [GLfloat]
   98 loadObj text =
   99     let verts = parseVertices text
  100         norms = parseNormals text
  101         texs = parseTextures text
  102         faces = parseFaces text
  103     in packObj verts norms texs faces
  104 
  105 -- | Given DLists containg raw information read from a
  106 --   .obj file, put together a Vec3 [GLfloat] of the
  107 --   vertices, texture coordinates, and normals,
  108 --   in that order, formatted to be sent to OpenGL.
  109 packObj ::
  110     D.DList GLfloat ->      -- ^ Vertices
  111     D.DList GLfloat ->      -- ^ Normals
  112     D.DList GLfloat ->      -- ^ Texture coords
  113     D.DList (Maybe Int) ->  -- ^ Face Definitions
  114     Vec3 [GLfloat]
  115 packObj verts norms texs faces =
  116     let lFaces = D.toList faces
  117         lVerts = V.fromList $ D.toList verts
  118         lNorms = V.fromList $ D.toList norms
  119         lTexs = V.fromList $ D.toList texs
  120 
  121         faceVerts = takeEvery3 1 lFaces
  122         faceTexs = takeEvery3 2 lFaces
  123         faceNorms = takeEvery3 3 lFaces
  124 
  125         realVerts = D.toList $ getIndicesD lVerts faceVerts 3
  126         realNorms = D.toList $ getIndicesD lNorms faceNorms 3
  127         realTexs = D.toList $ getIndicesD lTexs faceTexs 2
  128     in Vec3 realVerts realTexs realNorms
  129 
  130 -- | Given a List of () indices to read and the
  131 --   Vector to read from, read given indices for all
  132 --   Just values, and use -1.0 as a default value for
  133 --   Nothings. Returns a DList of GLfloats.
  134 getIndicesD :: V.Vector GLfloat -> [Maybe Int] -> Int -> D.DList GLfloat
  135 getIndicesD haystack (x:xs) vecType
  136     | isJust x =
  137             -- Get the section of the vector that is needed.
  138         let index = vecType * (fromJust x - 1)
  139             splitList = snd $ V.splitAt index haystack
  140             -- Switch to dlist for its O(1) appending.
  141             ret = D.fromList $ V.toList $ V.take vecType splitList
  142         in ret `D.append` getIndicesD haystack xs vecType
  143     | otherwise = D.replicate vecType (-1.0) `D.append` getIndicesD haystack xs vecType
  144 getIndicesD _ _ _ = D.empty
  145 
  146 takeEvery3 :: Int -> [a] -> [a]
  147 takeEvery3 offset (x:y:z:rest)
  148     | offset == 1 = x : takeEvery3 offset rest
  149     | offset == 2 = y : takeEvery3 offset rest
  150     | offset == 3 = z : takeEvery3 offset rest
  151     | otherwise = error $ "ObjLoader.takeEvery 3: arg"
  152                         ++ "must be 1, 2, or 3"
  153 takeEvery3 _ _ = []
  154 {-# INLINE takeEvery3 #-}
  155 
  156 parseFaces :: String -> D.DList (Maybe Int)
  157 parseFaces = D.concat . map faceLineCheck . lines
  158     where
  159     faceLineCheck line =
  160         if "f " `isPrefixOf` line
  161             then parseFaceLine line
  162         else D.empty
  163 
  164 parseFaceLine :: String -> D.DList (Maybe Int)
  165 parseFaceLine =
  166     D.concat . map parseFaceGroup . tail . words
  167 {-# INLINE parseFaceLine #-}
  168 
  169 parseFaceGroup :: String -> D.DList (Maybe Int)
  170 parseFaceGroup =
  171     D.fromList . map retrieveData . splitOn "/"
  172     where
  173     retrieveData text =
  174         if null text
  175             then Nothing
  176         else Just $ read text
  177 {-# INLINE parseFaceGroup #-}
  178 
  179 parseVertices :: String -> D.DList GLfloat
  180 parseVertices = parsePrefix "v "
  181 parseNormals :: String -> D.DList GLfloat
  182 parseNormals = parsePrefix "vn "
  183 parseTextures :: String -> D.DList GLfloat
  184 parseTextures =
  185     D.concat . map lineCheck . lines
  186     where
  187     lineCheck line =
  188         if "vt " `isPrefixOf` line
  189             then
  190                 let [x, y] = D.toList $ parseLine line
  191                 in D.fromList [x, 1-y]
  192         else D.empty
  193 
  194 -- | Call parseLine on all lines with the given prefix
  195 --   in given String.
  196 parsePrefix :: Read a => String -> String -> D.DList a
  197 parsePrefix prefix =
  198     D.concat . map lineCheck . lines
  199     where
  200     lineCheck line =
  201         if prefix `isPrefixOf` line
  202             then parseLine line
  203         else D.empty
  204     {-# INLINE lineCheck #-}
  205 
  206 parseLine :: Read a => String -> D.DList a
  207 parseLine = D.map read . D.tail . D.fromList . words
  208 {-# INLINE parseLine #-}

  210 loadObjMaterials :: String -> IO ([Material], [Material])
  211 loadObjMaterials contents = do
  212     library <- loadObjMaterialLib contents
  213 
  214     let listRet = listOfMats contents library emptyMaterial
  215     return (listRet, library)
  216 
  217 loadObjMaterialLib :: String -> IO [Material]
  218 loadObjMaterialLib =
  219     liftM concat . mapM loadMtlFile . parseMtlLibs
  220 {-# INLINE loadObjMaterialLib #-}
  221 
  222 parseMtlLibs :: String -> [FilePath]
  223 parseMtlLibs = getMaterialLibs . lines
  224 
  225 getMaterialLibs :: [String] -> [FilePath]
  226 getMaterialLibs (line:rest) =
  227     if "mtllib " `isPrefixOf` line
  228         then (last . words $ line) :
  229                 getMaterialLibs rest
  230     else getMaterialLibs rest
  231 getMaterialLibs _ = []
  232 {-# INLINE getMaterialLibs #-}
  233 
  234 listOfMats :: String -> [Material] -> Material -> [Material]
  235 listOfMats contents =
  236     lineAction (lines contents)
  237     where
  238     lineAction :: [String] -> [Material] -> Material -> [Material]
  239     lineAction (line:rest) library currentMat
  240         | "usemtl " `isPrefixOf` line =
  241             let mat = findMaterial (last . words $ line) library
  242             in lineAction rest library mat
  243         | "f " `isPrefixOf` line =
  244             replicate 3 currentMat ++
  245                 lineAction rest library currentMat
  246         | otherwise = lineAction rest library currentMat
  247     lineAction _ _ _ = []
  248 
  249 allMtlsInObj :: String -> [FilePath]
  250 allMtlsInObj =
  251     map (last . filter (not . null) . splitOn " ") . filter (isPrefixOf "mtllib ") . lines
  252 
  253 findMaterial :: String -> [Material] -> Material
  254 findMaterial name library = head $ filter (\x -> matName x == name) library
  255 {-# INLINE findMaterial #-}

  257 toArrays :: Vec3 [a] -> [[a]]
  258 toArrays (Vec3 x y z) = [x, y, z]
  259 
  260 fromVec3M :: [Maybe (Vec3 a)] -> [a]
  261 fromVec3M (Just (Vec3 x y z) : xs) =
  262     [x, y, z] ++ fromVec3M xs
  263 fromVec3M [] = []
  264 fromVec3M (Nothing : _) = error "fromVec3M: argument contained Nothing."
  265 
  266 fromJustSafe :: Num a => Maybe a -> a
  267 fromJustSafe (Just x) = x
  268 fromJustSafe Nothing = 0
  269 {-# INLINE fromJustSafe #-}
