   25 updateWorld :: Game t (World t)
   26 updateWorld = do
   27     world <- get
   28     newObjs <- updateEntities
   29     return world{worldEntities = newObjs}
   30     -- $ performUpdateAll world (worldEntities world)
   31 
   32 updateEntities :: Game t [GameObject t]
   33 updateEntities = do
   34     world <- get
   35     mapM (\o -> entityUpdate o o) (worldEntities world)

   37 -- | Safely move an object down, simulating
   38 --   very simple gravity.
   39 applyGravity :: World t -> GameObject t -> GameObject t
   40 applyGravity world object =
   41     moveObjectSafe world object (Vec3 0 (-0.1) 0)

   43 -- | Move an object on each axis independently, and return
   44 --   the first AABB it intersected with, if applicable.
   45 moveObjectSlideIntersecter ::
   46     World t -> GameObject t -> Vec3 GLfloat -> (GameObject t, Maybe AABB)
   47 moveObjectSlideIntersecter world object (Vec3 dx dy dz) =
   48     let (objectX, abX) = if dx /= 0
   49             then
   50                 let objectXP = moveObject object $ Vec3 dx 0 0
   51                     entities = findNearby (worldOctree world) (movePos playerAABB (getPos object))--worldEntities world
   52                     intersectingX = getObjectIntersecter objectXP entities
   53                 in if isJust intersectingX
   54                         then (object, intersectingX)
   55                     else (objectXP, Nothing)
   56             else (object, Nothing)
   57 
   58         (objectY, abY) = if dy /= 0
   59             then
   60                 let objectYP = moveObject objectX $ Vec3 0 dy 0
   61                     entities = findNearby (worldOctree world) (movePos playerAABB (getPos object))--worldEntities world
   62                     intersectingY = getObjectIntersecter objectYP entities
   63                 in if isJust intersectingY
   64                         then (objectX, intersectingY)
   65                     else (objectYP, abX)
   66             else (objectX, abX)
   67 
   68     in if dz /= 0
   69         then
   70             let objectZP = moveObject objectY $ Vec3 0 0 dz
   71                 entities = findNearby (worldOctree world) (movePos playerAABB (getPos object))--worldEntities world
   72                 intersectingZ = getObjectIntersecter objectZP entities
   73             in if isJust intersectingZ
   74                         then (objectY, intersectingZ)
   75                     else (objectZP, abY)
   76             else (objectY, abY)

   96                     entities = findNearby (worldOctree world) (movePos playerAABB (getPos object)) --worldEntities world
   97                     intersectingY = getObjectIntersecter objectYP entities
   98                 in if isJust intersectingY

  106                 entities = findNearby (worldOctree world) (movePos playerAABB (getPos object))--worldEntities world

  108             in if isJust intersectingZ
  109                         then
  110                             let filtered = filter isJust [abX, abY, intersectingZ]
  111                             in (objectY, if null filtered
  112                                             then Nothing
  113                                         else Just $ map fromJust filtered)
  114                     else
  115                         let filtered = filter isJust [abX, abY]
  116                         in (objectZP, if null filtered
  117                                         then Nothing
  118                                     else Just $ map fromJust filtered)
  119             else
  120                 let filtered = filter isJust [abX, abY]
  121                 in (objectY, if null filtered
  122                                 then Nothing
  123                             else Just $ map fromJust filtered)

  126 moveObjectSlide :: World t -> GameObject t -> Vec3 GLfloat -> GameObject t
  127 moveObjectSlide world object (Vec3 dx dy dz) =
  128     let objectX = if dx /= 0
  129         then
  130             let objectXP = moveObject object $ Vec3 dx 0 0
  131                 entities = findNearby (worldOctree world) (movePos playerAABB (getPos object)) --worldEntities world
  132                 intersectingX = isIntersectingAny objectXP entities
  133             in if intersectingX then object else objectXP
  134         else object
  135 
  136         objectY = if dy /= 0
  137             then
  138                 let objectYP = moveObject objectX $ Vec3 0 dy 0
  139                     entities = findNearby (worldOctree world) (movePos playerAABB (getPos object))--worldEntities world
  140                     intersectingY = isIntersectingAny objectYP entities
  141                 in if intersectingY then objectX else objectYP
  142             else objectX
  143 
  144     in if dz /= 0
  145         then
  146             let objectZP = moveObject objectY $ Vec3 0 0 dz
  147                 entities = findNearby (worldOctree world) (movePos playerAABB (getPos object))--worldEntities world
  148                 intersectingZ = isIntersectingAny objectZP entities
  149             in if intersectingZ then objectY else objectZP
  150         else objectY

  152 moveObjectSafe :: World t -> GameObject t -> Vec3 GLfloat -> GameObject t
  153 moveObjectSafe world object vec =
  154     let moved = moveObject object vec
  155         entities = findNearby (worldOctree world) (movePos playerAABB (getPos object))--worldEntities world
  156         intersectingObject = isIntersectingAny moved entities
  157     in if intersectingObject
  158         then object
  159     else moved

  161 moveObject :: GameObject t -> Vec3 GLfloat -> GameObject t
  162 moveObject obj deltaPos =
  163     setPos obj (getPos obj + deltaPos)
