   12 createModel ::
   13     FilePath ->     -- ^ Vertex Shader.
   14     FilePath ->     -- ^ Fragment Shader.
   15     [String] ->     -- ^ Attribute Variable names.
   16     [[GLfloat]] ->  -- ^ List containing all the lists of values.
   17                    --   (vertices, normals, etc).
   18     [GLuint] ->     -- ^ Size of each value.
   19     GLint ->        -- ^ Number of vertices.
   20     IO Model
   21 createModel vert frag attrNames buffData valLens vertCount = do
   22     program <- loadProgram vert frag
   23     attribs <- getAttrLocs program attrNames
   24     ids <- createBufferIdAll buffData
   25 
   26     let sAttribs = createShaderAttribs attribs ids valLens
   27     return $ Model (Shader program []) sAttribs [] vertCount
   28             (Just $ aabbByFace (head buffData))
   29             (Just $ aabbFromPoints (head buffData))
   30
