   19 data Material = Material {
   20     matName :: String,
   21     matAmbientColor :: Maybe (Vec3 GLfloat),
   22     matDiffuseColor :: Maybe (Vec3 GLfloat),
   23     matSpecularColor :: Maybe (Vec3 GLfloat),
   24     matTexture :: Maybe GL.TextureObject,
   25     matTexId :: Maybe GLint
   26 } deriving (Show)
   27 
   28 loadMtlFile :: FilePath -> IO [Material]
   29 loadMtlFile file =
   30     openFile file ReadMode >>= loadMtlMaterials
   31 
   32 loadMtlMaterials :: Handle -> IO [Material]
   33 loadMtlMaterials handle =
   34     liftM (map applyDefualtMtl . tail)
   35           (loadMtlMaterialsRec 0 handle emptyMaterial)
   36 
   37 -- | Apply defualt values to attributes set to
   38 --   Nothing according to spec at
   39 --   http://people.sc.fsu.edu/~jburkardt/data/mtl/mtl.html
   40 applyDefualtMtl :: Material -> Material
   41 applyDefualtMtl mat@(Material _ amb diff spec _ texId) =
   42     let newAmb = if isNothing amb
   43                     then Just $ Vec3 0.2 0.2 0.2
   44                 else amb
   45         newDiff = if isNothing diff
   46                     then Just $ Vec3 0.8 0.8 0.8
   47                 else diff
   48         newSpec = if isNothing spec
   49                     then Just $ Vec3 1.0 1.0 1.0
   50                 else spec
   51         newTexId = if isNothing texId
   52                     then Just (-1)
   53                 else texId
   54     in mat{matAmbientColor = newAmb,
   55            matDiffuseColor = newDiff,
   56            matSpecularColor = newSpec,
   57            matTexId = newTexId}
   58 
   59 -- | UNSAFE!! Use loadMtlMaterials instead.
   60 loadMtlMaterialsRec :: GLuint -> Handle -> Material -> IO [Material]
   61 loadMtlMaterialsRec textureCount handle start = do
   62     eof <- hIsEOF handle
   63     if not eof
   64         then do 
   65             line <- hGetLine handle
   66             -- If there is a declaration of a new Material,
   67             -- "add the current mat to the list" and start
   68             -- on a new Material.
   69             if "newmtl " `isPrefixOf` line
   70                 then do
   71                     let name = head $ rawMtlLine line
   72                     rest <- loadMtlMaterialsRec textureCount handle
   73                                 (Material name Nothing Nothing Nothing Nothing Nothing)
   74                     return $ start : rest
   75             else if not $ null line
   76                 -- Call executeCommand on current line and Material
   77                 -- and then continue adding attributes to that Material.
   78                 then
   79                     let origTex = matTexture start
   80                     in do
   81                         newMat <- executeCommand line start textureCount
   82                         if matTexture newMat == origTex
   83                             then loadMtlMaterialsRec textureCount handle newMat
   84                         else loadMtlMaterialsRec (textureCount+1) handle newMat
   85             else loadMtlMaterialsRec textureCount handle start
   86     -- If it is End Of File, close the file and
   87     -- return the last Material.
   88     else hClose handle >> return [start]
   89 
   90 executeCommand :: String -> Material -> GLuint -> IO Material
   91 executeCommand command mat textureCount
   92     | "Ka " `isPrefixOf` command =
   93         return mat{matAmbientColor = Just $ readMtlLineTriplet command}
   94     | "Kd " `isPrefixOf` command =
   95         return mat{matDiffuseColor = Just $ readMtlLineTriplet command}
   96     | "Ks " `isPrefixOf` command =
   97         return mat{matSpecularColor = Just $ readMtlLineTriplet command}
   98     | "map_Kd " `isPrefixOf` command = do
   99         texture <- juicyLoadTexture $ head (rawMtlLine command)
  100         return mat{matTexture = Just texture,
  101                    matTexId = Just $ fromIntegral textureCount}
  102     | otherwise = return mat

  104 allImagesInFile :: String -> [FilePath]
  105 allImagesInFile =
  106     map (last . filter (not . null) . splitOn " ") . filter (isPrefixOf "map_Kd ") . lines
  107 
  108 readMtlLineTriplet :: String -> Vec3 GLfloat
  109 readMtlLineTriplet = toTripletMtl . readMtlLine
  110 
  111 readMtlLine :: String -> [GLfloat]
  112 readMtlLine = map read . tail . filter (not . null) . splitOn " "
  113 
  114 rawMtlLine :: String -> [String]
  115 rawMtlLine = tail . filter (not . null) . splitOn " "
  116 
  117 toTripletMtl :: [a] -> Vec3 a
  118 toTripletMtl xs
  119     | length xs == 3 = Vec3 (head xs) (xs !! 1) (xs !! 2)
  120     | otherwise = error "Material.toTripletMtl"
  121 
  122 emptyMaterial :: Material
  123 emptyMaterial = Material "" Nothing Nothing Nothing Nothing Nothing
