   60 -- | Loads a pair of vertex and fragment shaders
   61 --   given the two FilePaths.
   62 loadProgram :: FilePath -> FilePath -> IO GLuint
   63 loadProgram vertFP fragFP =
   64     loadShadersProgram
   65         [(gl_VERTEX_SHADER,   vertFP),
   66          (gl_FRAGMENT_SHADER, fragFP)]
   67 
   68 loadProgramComplete ::
   69     FilePath -> FilePath -> FilePath -> FilePath -> FilePath -> IO GLuint
   70 loadProgramComplete vertFP fragFP tessCFP tessEFP geomFP =
   71     loadShadersProgram
   72         [(gl_VERTEX_SHADER, vertFP),
   73          (gl_FRAGMENT_SHADER, fragFP),
   74          (gl_TESS_CONTROL_SHADER, tessCFP),
   75          (gl_TESS_EVALUATION_SHADER, tessEFP),
   76          (gl_GEOMETRY_SHADER, geomFP)]

   78 loadShadersProgram :: [(GLuint, FilePath)] -> IO GLuint
   79 loadShadersProgram shaders = do
   80     shaderIds <- mapM (uncurry loadShader) shaders
   81     progId <- glCreateProgram
   82     mapM_ (glAttachShader progId) shaderIds
   83     glLinkProgram progId
   84     mapM_ glDeleteShader shaderIds
   85     return progId
   86 
   87 -- | Loads a single shader of given type,
   88 --   usually either gl_VERTEX_SHADER or
   89 --   gl_FRAGMENT_SHADER.
   90 --   Uses given FilePath as shader.
   91 loadShader :: GLenum -> FilePath -> IO GLuint
   92 loadShader shaderTypeFlag filePath = do
   93     code <- readFile filePath
   94     sid <- glCreateShader shaderTypeFlag
   95     withCString code $ \codePtr ->
   96         with codePtr $ \codePtrPtr ->
   97             glShaderSource sid 1 codePtrPtr nullPtr
   98     glCompileShader sid
   99 
  100     _ <- checkStatus gl_COMPILE_STATUS glGetShaderiv glGetShaderInfoLog sid
  101 
  102     return sid

  104 -- | Gets status from OpenGL. Supply flag to check,
  105 --   function to get status with, infoLog function,
  106 --   and shader id.
  107 checkStatus :: (Storable a1, Num a1, Eq a1, Ord a1, Integral a1) =>
  108     GLenum ->
  109     (t -> GLenum -> Ptr a1 -> IO a) ->
  110     (t -> a1 -> Ptr a3 -> Ptr CChar -> IO a2) ->
  111     t ->
  112     IO Bool
  113 checkStatus statusFlag glGetFn glInfoLogFn idT = do
  114     let fetch info = withNewPtr (glGetFn idT info)
  115     status <- toBool <$> fetch statusFlag
  116     logLength <- fetch gl_INFO_LOG_LENGTH
  117     when (logLength > 0) $
  118         allocaArray0 (fromIntegral logLength) $ \msgPtr -> do
  119             _ <- glInfoLogFn idT logLength nullPtr msgPtr
  120             peekCString msgPtr >>=
  121                 if status 
  122                     then \t -> do
  123                         putStr "Successfully loaded shader: "
  124                         print t
  125                 else \t -> do
  126                     putStr "Error loading shader: "
  127                     print t
  128     return status

  162 -- | Binds a list of ShaderAttribs.
  163 setShaderAttribs :: [ShaderAttrib] -> IO ()
  164 setShaderAttribs (Vec3 attr buf len : rest) = do
  165     -- Enable the attribute buffer.
  166     glEnableVertexAttribArray attr
  167     -- Give OpenGL the information.
  168     glBindBuffer gl_ARRAY_BUFFER buf
  169     -- Tell OpenGL about the info.
  170     glVertexAttribPointer attr (fromIntegral len) gl_FLOAT 0 0 GU.offset0
  171     setShaderAttribs rest
  172 setShaderAttribs [] = return ()
  173 
  174 -- | Disables a list of ShaderAttribs. Call after drawing?
  175 disableShaderAttribs :: [ShaderAttrib] -> IO ()
  176 disableShaderAttribs (Vec3 attr _ _ : rest) = do
  177     -- Disable the attribute buffer.
  178     glDisableVertexAttribArray attr
  179     disableShaderAttribs rest
  180 disableShaderAttribs [] = return ()

  182 setUniformsAndRemember :: Shader -> [ShaderUniform] -> IO Shader
  183 setUniformsAndRemember shader ((name, valsIo):rest) = do
  184     vals <- valsIo
  185     let len = length vals
  186     let uniloc = findMaybeUniformLocation shader name
  187     loc <- maybe (withCString name $ glGetUniformLocation $ shaderId shader) return uniloc
  188 
  189     case len of
  190         1 -> glUniform1f loc $ head vals
  191         2 -> glUniform2f loc (head vals) $ vals !! 1
  192         3 -> glUniform3f loc (head vals) (vals !! 1) (vals !! 2)
  193         4 -> glUniform4f loc (head vals) (vals !! 1) (vals !! 2) (vals !! 3)
  194         _ -> putStrLn $ "Bad length value in ShaderUniform " 
  195                         ++ name ++ ": " ++ show len
  196 
  197     if isJust uniloc
  198         then
  199             let newShader = shader{shaderUniforms =
  200                     (name, loc) : shaderUniforms shader}
  201             in setUniformsAndRemember newShader rest
  202     else setUniformsAndRemember shader rest
  203 setUniformsAndRemember shader _ = return shader
  204 
  205 setUniformAndRemember :: Shader -> ShaderUniform -> IO Shader
  206 setUniformAndRemember shader (name, valsIo) = do
  207     vals <- valsIo
  208     let len = length vals
  209     
  210     loc <- let uniloc = findMaybeUniformLocation shader name
  211           in maybe (withCString name $ glGetUniformLocation $ shaderId shader)
  212                     return uniloc
  213 {-
  214           in if isJust uniloc
  215                 then return $ fromJust uniloc
  216             else withCString name $ glGetUniformLocation $ shaderId shader
  217 -}
  218 
  219     case len of
  220         1 -> glUniform1f loc $ head vals
  221         2 -> glUniform2f loc (head vals) $ vals !! 1
  222         3 -> glUniform3f loc (head vals) (vals !! 1) (vals !! 2)
  223         4 -> glUniform4f loc (head vals) (vals !! 1) (vals !! 2) (vals !! 3)
  224         _ -> putStrLn $ "Bad length value in ShaderUniform " 
  225                         ++ name ++ ": " ++ show len
  226 
  227     return $ shader{shaderUniforms = (name, loc) : shaderUniforms shader}

  229 findUniformLocationAndRemember :: Shader -> String -> IO (Shader, GLint)
  230 findUniformLocationAndRemember shader name =
  231     let found = findMaybeUniformLocation shader name
  232     in if isJust found
  233             then return (shader, fromJust found)
  234         else do
  235             loc <- quickGetUniform (shaderId shader) name
  236             return (shader{
  237                 shaderUniforms =
  238                     (name, loc) : shaderUniforms shader
  239             }, loc)

  241 findUniformLocation :: Shader -> String -> IO GLint
  242 findUniformLocation shader name =
  243     let found = findMaybeUniformLocation shader name
  244     in maybe (quickGetUniform (shaderId shader) name) return found
  245 
  246 findMaybeUniformLocation :: Shader -> String -> Maybe GLint
  247 findMaybeUniformLocation shader =
  248     findUniformLocation' (shaderUniforms shader)
  249     where
  250         findUniformLocation' ((curName, curId):rest) searchName =
  251             if curName == searchName
  252                 then Just curId
  253             else findUniformLocation' rest searchName
  254         findUniformLocation' _ _ = Nothing

  276 -- | Retrieve location of each shader attrib
  277 --   in the given program.
  278 getAttrLocs :: GLuint -> [String] -> IO [GLuint]
  279 getAttrLocs prog (attrName:xs) = do
  280     curN <- withCString attrName $ glGetAttribLocation prog
  281     let cur = fromIntegral curN
  282     rest <- getAttrLocs prog xs
  283     return $ cur:rest
  284 getAttrLocs _ [] = return []
