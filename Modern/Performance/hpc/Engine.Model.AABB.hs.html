   18 -- | Test if two objects intersect.
   19 objectsIntersect :: (HasAABB a, HasAABB b) => a -> b -> Bool
   20 objectsIntersect l r
   21     | isJust (getWholeAABB l) &&
   22       isJust (getWholeAABB r) =
   23         let Just wholeabl = calculateNewWholeAABB l
   24             Just wholeabr = calculateNewWholeAABB r
   25         in intersecting wholeabl wholeabr &&
   26             (null (getAABBs l) && (not . null) (getAABBs r) ||
   27                 let newl = calculateNewAABBs l
   28                     newr = calculateNewAABBs r
   29                 in anyIntersect (head newl) newr)
   30     | otherwise =
   31         null (getAABBs l) && (not . null) (getAABBs r) ||
   32             let newl = calculateNewAABBs l
   33                 newr = calculateNewAABBs r
   34             in anyIntersect (head newl) newr
   35 
   36 -- | Check if given point is inside the AABB.
   37 aabbContainsPoint :: Vec3 GLfloat -> AABB -> Bool
   38 aabbContainsPoint (Vec3 x y z) (AABB (Vec3 lx ly lz) (Vec3 hx hy hz)) =
   39     x >= lx && x <= hx &&
   40     y >= ly && y <= hy &&
   41     z >= lz && z <= hz
   42 
   43 -- | Using the object's current AABB and position,
   44 --   create a new AABB.
   45 calculateNewWholeAABB :: HasAABB a => a -> Maybe AABB
   46 calculateNewWholeAABB obj
   47     | isJust (getWholeAABB obj) =
   48         let (Just (AABB l r)) = getWholeAABB obj
   49             pos = getPos obj
   50         in Just $ AABB (l + pos) (r + pos)
   51     | otherwise = Nothing
   52 
   53 -- | Using the object's current AABBs and position,
   54 --   create a new AABB.
   55 calculateNewAABBs :: HasAABB a => a -> [AABB]
   56 calculateNewAABBs obj
   57     | (not . null) (getAABBs obj) =
   58         let aabbs = getAABBs obj
   59         in transformAll aabbs (getPos obj)
   60     | otherwise = []
   61 
   62 transformAll :: [AABB] -> Vec3 GLfloat -> [AABB]
   63 transformAll (AABB l r : xs) pos =
   64     AABB (l + pos) (r + pos) : transformAll xs pos
   65 transformAll [] _ = []
   66 
   67 -- | Check if the needle intersects with any in the
   68 --   haystack.
   69 isIntersectingAny :: (HasAABB a, HasAABB b) => a -> [b] -> Bool
   70 isIntersectingAny collider (collidee:xs) =
   71     objectsIntersect collider collidee ||
   72         isIntersectingAny collider xs
   73 isIntersectingAny _ [] = False
   74 
   75 -- | Check if the needle intersects with any in the haystack,
   76 --   if it does, the intersected AABB is returned.
   77 getObjectIntersecter :: (HasAABB a, HasAABB b) => a -> [b] -> Maybe AABB
   78 getObjectIntersecter collider (collidee:xs) =
   79     let intersecter = getIntersecter collider collidee
   80     in if isJust intersecter
   81         then intersecter
   82     else getObjectIntersecter collider xs
   83 getObjectIntersecter _ [] = Nothing
   84 
   85 -- | Test if two objects intersect, yeilding the
   86 --   offending AABB if they do.
   87 getIntersecter :: (HasAABB a, HasAABB b) => a -> b -> Maybe AABB
   88 getIntersecter l r
   89     | isJust (getWholeAABB l) &&
   90       isJust (getWholeAABB r) =
   91         let Just wholeabl = calculateNewWholeAABB l
   92             Just wholeabr = calculateNewWholeAABB r
   93         in
   94             if intersecting wholeabl wholeabr
   95                 then if null (getAABBs l) && (not . null) (getAABBs r)
   96                     then Just wholeabr
   97                 else
   98                     let newl = calculateNewAABBs l
   99                         newr = calculateNewAABBs r
  100                     in anyIntersectGet (head newl) newr
  101             else Nothing
  102     | otherwise = Nothing

  105 -- | Takes a Vec3 containing the length, height, and width
  106 --   and returns a Vec3 with those dimensions.
  107 createAABB :: Vec3 GLfloat -> AABB
  108 createAABB = AABB (Vec3 0 0 0)
  109 {-# INLINE createAABB #-}

  111 -- | Creates an aabb for every n faces.
  112 aabbsFromPointsGrouped :: [GLfloat] -> Int -> [AABB]
  113 aabbsFromPointsGrouped points groupSize
  114     | length points >= groupSize =
  115         let (cur, rest) = splitAt (groupSize*3) points
  116             xs = getXs cur
  117             ys = getYs cur
  118             zs = getZs cur
  119             maxX = minimum xs
  120             maxY = minimum ys
  121             maxZ = minimum zs
  122             minX = minimum xs
  123             minY = minimum ys
  124             minZ = minimum zs
  125            
  126         in AABB (Vec3 minX minY minZ) (Vec3 maxX maxY maxZ) :
  127                 aabbsFromPointsGrouped rest groupSize
  128     | otherwise = []
  129 
  130 -- | Calculate an AABB for all the points.
  131 aabbFromPoints :: [GLfloat] -> AABB
  132 aabbFromPoints points@(x:y:z:_) =
  133     aabbFromPointsAccum points (Vec3 x y z) (Vec3 x y z)
  134 aabbFromPoints _ =
  135     error $ "AABB.aabbFromPoints: given list is too short or " ++
  136             "number of elements is not divisible by 3."
  137 
  138 -- | Called by aabbFromPoints.
  139 aabbFromPointsAccum :: [GLfloat] -> Vec3 GLfloat -> Vec3 GLfloat -> AABB
  140 aabbFromPointsAccum (x:y:z:rest) (Vec3 miX miY miZ) (Vec3 maX maY maZ) =
  141     aabbFromPointsAccum rest
  142         (Vec3 (min miX x) (min miY y) (min miZ z))
  143         (Vec3 (max maX x) (max maY y) (max maZ z))
  144 aabbFromPointsAccum _ abMin abMax = AABB abMin abMax
  145 
  146 -- | Calculate an AABB for each triangulated
  147 --   face.
  148 aabbByFace :: [GLfloat] -> [AABB]
  149 aabbByFace (x1:y1:z1:x2:y2:z2:x3:y3:z3:rest) =
  150     let minVec = Vec3 (min3 x1 x2 x3) (min3 y1 y2 y3) (min3 z1 z2 z3)
  151         maxVec = Vec3 (max3 x1 x2 x3) (max3 y1 y2 y3) (max3 z1 z2 z3)
  152     in AABB minVec maxVec : aabbByFace rest
  153 aabbByFace [] = []
  154 aabbByFace _ =
  155     error "AABB.aabbByFace: given list is not formatted properly."
  156 
  157 -- | Move an AABB by adding given Vec3 to the min
  158 --   and max points.
  159 moveAABB :: AABB -> Vec3 GLfloat -> AABB
  160 moveAABB (AABB origMin origMax) deltaMovement =
  161     AABB (origMin + deltaMovement) (origMax + deltaMovement)
  162 {-# INLINE moveAABB #-}
  163 
  164 -- | Calculate the width, height and length of an AABB.
  165 aabbDimensions :: AABB -> Vec3 GLfloat
  166 aabbDimensions (AABB left right) = right - left
  167 {-# INLINE aabbDimensions #-}

  169 -- | Check if two Maybe AABB's are intersecting.
  170 --   if either are Nothing, it is False.
  171 mIntersecting :: Maybe AABB -> Maybe AABB -> Bool
  172 mIntersecting (Just left) (Just right) = intersecting left right
  173 mIntersecting Nothing Nothing = False
  174 mIntersecting Nothing _ = False
  175 mIntersecting _ Nothing = False
  176 
  177 -- | Check if the given AABB is intersecting with
  178 --   any of the others.
  179 anyIntersect :: AABB -> [AABB] -> Bool
  180 anyIntersect l (r:rs) =
  181     intersecting l r || anyIntersect l rs
  182 anyIntersect _ _ = False
  183 
  184 anyIntersectGet :: AABB -> [AABB] -> Maybe AABB
  185 anyIntersectGet l (r:rs) =
  186     if intersecting l r
  187         then Just r
  188     else anyIntersectGet l rs
  189 anyIntersectGet _ _ = Nothing
  190 
  191 -- | Check if 2 AABB's are intersecting.
  192 intersecting :: AABB -> AABB -> Bool
  193 intersecting (AABB (Vec3 min1x min1y min1z) (Vec3 max1x max1y max1z))
  194              (AABB (Vec3 min2x min2y min2z) (Vec3 max2x max2y max2z)) =
  195     max1x >= min2x &&
  196     min1x <= max2x &&
  197     max1y >= min2y &&
  198     min1y <= max2y &&
  199     max1z >= min2z &&
  200     min1z <= max2z
  201 {-# INLINE intersecting #-}
  202 
  203 getXs :: [a] -> [a]
  204 getXs (x:_:_:rest) =
  205     x : getXs rest
  206 getXs _ = []
  207 
  208 getYs :: [a] -> [a]
  209 getYs (_:y:_:rest) =
  210     y : getYs rest
  211 getYs _ = []
  212 
  213 getZs :: [a] -> [a]
  214 getZs (_:_:z:rest) =
  215     z : getZs rest
  216 getZs _ = []

  218 min3 :: Ord a => a -> a -> a -> a
  219 min3 a b c = min c $ min a b
  220 {-# INLINE min3 #-}
  221 
  222 max3 :: Ord a => a -> a -> a -> a
  223 max3 a b c = max c $ max a b
  224 {-# INLINE max3 #-}
