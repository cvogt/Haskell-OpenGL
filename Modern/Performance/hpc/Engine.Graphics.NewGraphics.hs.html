   18 data RenderInfo = RenderInfo {
   19     renderInfoShader :: Shader,
   20     renderInfoMatrices :: WorldMatrices
   21 }
   22 
   23 emptyInfo :: RenderInfo
   24 emptyInfo = RenderInfo emptyShader emptyMatrices
   25 
   26 -- | A class for things that can be rendered to
   27 --   the screen &| Framebuffers.
   28 --   Contains 3 functions:
   29 --      Bind:
   30 --          This is usually where shaders are set,
   31 --          or the currently bound shader is
   32 --          modified.
   33 --      Draw:
   34 --          This is the draw action. A Framebuffer
   35 --          is bound before calling this, so there is
   36 --          no need to bind your own.
   37 --      Cleanup:
   38 --          Any cleanup necessary.
   39 class Renderable t g where
   40     renderBind :: t -> g -> IO g
   41     renderBind _ = return
   42     renderDraw :: t -> g -> IO g
   43     renderDraw _ = return
   44     renderCleanup :: t -> g -> IO g
   45     renderCleanup _ = return
   46     defaultGlobal :: t -> g
   47 
   48 instance Renderable (GameObject t) RenderInfo where
   49     renderBind obj info =
   50         let shader = modelShader $ getModel obj
   51         in do
   52             glUseProgram $ shaderId shader
   53             return info{renderInfoShader = shader}
   54 
   55     renderDraw object info = do
   56         let model = getModel object
   57             Vec3 objx objy objz = getPos object
   58             Vec3 objrx objry objrz = getRot object
   59             mShader = renderInfoShader info
   60 
   61             -- Move Object
   62             modelMat = gtranslationMatrix [objx, objy, objz] *
   63                        grotationMatrix [objrx, objry, objrz]
   64 
   65             newMatrices = (renderInfoMatrices info){matrixModel = modelMat}
   66         -- Set uniforms. (World uniforms and Matrices).
   67         newShader <-
   68             setMatrixUniforms mShader newMatrices
   69 
   70         -- Bind buffers to variable names in shader.
   71         setShaderAttribs $ modelShaderVars model
   72         bindTextures (modelTextures model) $ shaderId newShader
   73 
   74         glDrawArrays gl_TRIANGLES 0 (modelVertCount model)
   75 
   76         return $ RenderInfo newShader newMatrices
   77 
   78     renderCleanup object _ = do
   79         let model = getModel object
   80         -- Necessary?
   81         disableShaderAttribs $ modelShaderVars model
   82         -- Disable the object's shader.
   83         glUseProgram 0
   84         return emptyInfo
   85     defaultGlobal _ = emptyInfo
   86 
   87 instance Renderable (World t) RenderInfo where
   88     renderBind world info = do
   89         let (winW, winH) = windowSize $ stateWindow $ worldState world
   90             worldMats = calculateMatricesFromPlayer
   91                         (worldPlayer world) (fromIntegral winW, fromIntegral winH)
   92         newShader <-
   93             setWorldUniforms world (renderInfoShader info)
   94         return info{renderInfoShader = newShader,
   95                     renderInfoMatrices = worldMats}
   96     defaultGlobal _ = emptyInfo
   97         
   98 
   99 totalRender :: Renderable t g => t -> g -> IO g
  100 totalRender r s =
  101     renderBind r s >>=
  102     renderDraw r >>=
  103     renderCleanup r
  104 
  105 
  106 renderToFramebuffer :: forall t g. Renderable t g => Framebuffer -> t -> IO g
  107 renderToFramebuffer fbuf rend =
  108     withFramebuffer fbuf $ totalRender rend (defaultGlobal rend)
  109 
  110 renderAllToFramebuffer :: forall t g. Renderable t g => Framebuffer -> [t] -> IO g
  111 renderAllToFramebuffer fbuf xs =
  112     withFramebuffer fbuf $
  113         renderAll xs (defaultGlobal $ head xs)
  114     where
  115         renderAll :: Renderable t g => [t] -> g -> IO g
  116         renderAll (x:xs') info =
  117             totalRender x info >>= renderAll xs'
  118 
  119 renderAllWithGlobal :: (Renderable t1 g, Renderable t2 g) =>
  120                         Framebuffer -> t1 -> [t2] -> IO g
  121 renderAllWithGlobal fbuf g =
  122     renderAllWithGlobal' (defaultGlobal g) fbuf g
  123 
  124 renderAllWithGlobal' :: (Renderable t1 g, Renderable t2 g) =>
  125                         g -> Framebuffer -> t1 -> [t2] -> IO g
  126 renderAllWithGlobal' info fbuf global (x:xs) = do
  127     bindFrameBuffer fbuf
  128 
  129     newinfo <-
  130         renderBind x info >>=
  131         renderBind global >>=
  132         renderDraw global >>=
  133         renderDraw x >>=
  134         renderCleanup x >>=
  135         renderCleanup global
  136 
  137     unbindFrameBuffer
  138 
  139     renderAllWithGlobal' newinfo fbuf global xs
  140 renderAllWithGlobal' info _ _ _ = return info
  141

  143 renderWorldNew :: World t -> IO (World t)
  144 renderWorldNew world = do
  145     glClear $ gl_COLOR_BUFFER_BIT .|. gl_DEPTH_BUFFER_BIT
  146     renderAllWithGlobal (screenFramebuffer (800, 600)) world (worldEntities world) :: IO RenderInfo
  147     return world
  148 
  149 renderWorldNewWithFramebuffer :: World t -> Framebuffer -> IO (World t)
  150 renderWorldNewWithFramebuffer world fbuf = do
  151     glClear $ gl_COLOR_BUFFER_BIT .|. gl_DEPTH_BUFFER_BIT
  152     renderAllWithGlobal fbuf world (worldEntities world) :: IO RenderInfo
  153     return world

  156 -- | Render world with all postprocessing shaders defined by
  157 --   worldFramebuffer.
  158 renderWorldNewPost :: World t -> IO (World t)
  159 renderWorldNewPost world = do
  160     let effects = snd $ graphicsPostProcessors $ worldGraphics world
  161         fb = fst $ graphicsPostProcessors $ worldGraphics world 
  162     glBindFramebuffer gl_FRAMEBUFFER $
  163         fbufName fb
  164     ret <- renderWorldNewWithFramebuffer world fb
  165     bindFrameBuffer fb
  166 
  167     renderAllPasses ret effects
  168     return ret
  169

  171 screenFramebuffer :: (GLint, GLint) -> Framebuffer
  172 screenFramebuffer dimensions =
  173     FB 0 0 dimensions 0 0
  174 
  175 withFramebuffer :: Framebuffer -> IO a -> IO a
  176 withFramebuffer fbuf func = do
  177     bindFrameBuffer fbuf
  178     ret <- func
  179     unbindFrameBuffer
  180     return ret
  181 
  182 bindFrameBuffer :: Framebuffer -> IO ()
  183 bindFrameBuffer = glBindFramebuffer gl_FRAMEBUFFER . fbufName
  184 
  185 unbindFrameBuffer :: IO ()
  186 unbindFrameBuffer = glBindFramebuffer gl_FRAMEBUFFER 0
