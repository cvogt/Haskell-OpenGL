1 {-# LANGUAGE RankNTypes #-}
    2 module Engine.Graphics.GraphicsUtils (
    3     createBufferIdAll, createBufferId,
    4     fillNewBuffer, withNewPtr, withNewPtrArray,
    5     useNewPtr
    6 ) where
    7 
    8 import Foreign
    9     (Ptr, Storable, withArrayLen, sizeOf,
   10      alloca, peek, peekArray)
   11 
   12 import Graphics.Rendering.OpenGL.Raw
   13 
   14 -- | Create an id for each buffer data.
   15 createBufferIdAll :: [[GLfloat]] -> IO [GLuint]
   16 createBufferIdAll (cur:others) = do
   17     currentId <- createBufferId cur
   18     otherId <- createBufferIdAll others
   19     return $ currentId:otherId
   20 createBufferIdAll [] = return []
   21 
   22 -- | Create a buffer id for the information.
   23 createBufferId :: [GLfloat] -> IO GLuint
   24 createBufferId info = do
   25     vertexArrayId <- withNewPtr (glGenVertexArrays 1)
   26     glBindVertexArray vertexArrayId
   27     fillNewBuffer info
   28 
   29 -- | Fill buffer with data.
   30 fillNewBuffer :: [GLfloat] -> IO GLuint
   31 fillNewBuffer list = do
   32     bufId <- withNewPtr (glGenBuffers 1)
   33     glBindBuffer gl_ARRAY_BUFFER bufId
   34     withArrayLen list $ \len ptr ->
   35         glBufferData gl_ARRAY_BUFFER
   36             (fromIntegral (len * sizeOf (undefined :: GLfloat)))
   37             (ptr :: Ptr GLfloat) gl_STATIC_DRAW
   38     return bufId
   39 
   40 -- | Perform IO action with a new pointer, returning the
   41 --   value in the pointer.
   42 withNewPtr :: forall b a. Storable b => (Ptr b -> IO a) -> IO b
   43 withNewPtr f = alloca (\p -> f p >> peek p)
   44 
   45 -- | Perform IO action with a new pointer array, returning the
   46 --   value in the pointer.
   47 withNewPtrArray :: forall b a. Storable b => (Ptr b -> IO a) -> Int -> IO [b]
   48 withNewPtrArray f size = alloca (\p -> f p >> peekArray size p)
   49 
   50 -- | Perform IO action with a new pointer, returning the
   51 --   pointer itself.
   52 useNewPtr :: forall a a1. Storable a => (Ptr a -> IO a1) -> IO (Ptr a)
   53 useNewPtr f = alloca (\p -> f p >> return p)
