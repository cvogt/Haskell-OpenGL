1 module Engine.Graphics.Window (
    2     defaultWindow, Window(..),
    3     openWindow, shutdown
    4 ) where
    5 
    6 import System.Exit (exitSuccess)
    7 
    8 import qualified Graphics.UI.GLFW as GLFW
    9 
   10 data Window = Window {
   11     windowHints :: [GLFW.WindowHint],
   12     windowTitle :: String,
   13     windowSize :: (Int, Int),
   14     windowInner :: Maybe GLFW.Window
   15 }
   16 
   17 defaultWindow :: Window
   18 defaultWindow =
   19     Window
   20         [GLFW.WindowHint'RefreshRate 60,
   21          GLFW.WindowHint'Samples 4]
   22         "GLFW Window"
   23         (800, 600)
   24         Nothing
   25 
   26 openWindow :: Window -> IO Window
   27 openWindow window =
   28     let (w, h) = windowSize window 
   29     in do
   30     _ <- GLFW.init
   31 
   32     -- Apply window hints.
   33     mapM_ GLFW.windowHint $ windowHints window
   34 
   35     Just win <- GLFW.createWindow w h (windowTitle window) Nothing Nothing
   36     GLFW.setWindowPos win 100 50
   37     GLFW.makeContextCurrent (Just win)
   38 
   39     -- Enable VSync
   40     GLFW.swapInterval 1
   41 
   42     --GLFW.setWindowCloseCallback win (Just shutdown)
   43 
   44     return $ window{windowInner = Just win}
   45 
   46 shutdown :: GLFW.WindowCloseCallback
   47 shutdown win = do
   48     GLFW.destroyWindow win
   49     GLFW.terminate
   50     _ <- exitSuccess
   51     return ()
