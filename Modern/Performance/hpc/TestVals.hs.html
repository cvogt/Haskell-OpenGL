   71 mkWorld :: Framebuffer -> Framebuffer -> GLuint -> [GLuint] -> IO (World ())
   72 mkWorld fb shadowFb shadowShader shaders = do
   73     state <- mkWorldState
   74     obja <- mkObj
   75     objb <- mkObj2
   76     objc <- mkObj3
   77     let octree = mkOctree [obja, objb, objc]
   78 
   79     --print octree
   80 
   81     return $ World mkPlayer [obja, objb, objc] octree
   82             (mkGraphics fb shadowFb shadowShader shaders)
   83             state
