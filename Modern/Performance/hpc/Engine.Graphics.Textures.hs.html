   27 -- | Load an image and turn it into something OpenGL can use.
   28 juicyLoadTexture :: FilePath -> IO GL.TextureObject
   29 juicyLoadTexture file = do
   30     (Image (Size w h) pd) <- juicyLoadImageRaw file
   31     texName <- liftM head (GL.genObjectNames 1)
   32     GL.textureBinding GL.Texture2D $= Just texName
   33     GL.textureFilter GL.Texture2D $= ((GL.Nearest, Nothing), GL.Nearest)
   34     GL.texImage2D GL.Texture2D GL.NoProxy
   35         0
   36         GL.RGB' (GL.TextureSize2D w h) 0 pd
   37     return texName
   38 
   39 -- TODO: add support for all (most) colorspaces / formats.
   40 juicyLoadImageRaw :: FilePath -> IO Image
   41 juicyLoadImageRaw file = do
   42     image <- Juicy.readImage file
   43 
   44     case image of
   45         Left err -> error err
   46         Right (Juicy.ImageRGB8 (Juicy.Image w h dat)) ->
   47             unsafeWith dat $ \ptr ->
   48             return $ Image (GL.Size (fromIntegral w) (fromIntegral h))
   49                             (GL.PixelData GL.RGB UnsignedByte ptr)
   50         Right (Juicy.ImageYCbCr8 img) ->
   51             let (Juicy.Image w h dat) =
   52                     JTypes.convertImage img :: Juicy.Image Juicy.PixelRGB8
   53             in unsafeWith dat $ \ptr ->
   54                 return $ Image (GL.Size (fromIntegral w) (fromIntegral h))
   55                             (GL.PixelData GL.RGB UnsignedByte ptr)
   56         _ -> error $
   57             "Engine.Graphics.Texture.juicyLoadImage:"
   58                 ++ "bad image colorspace or format."
