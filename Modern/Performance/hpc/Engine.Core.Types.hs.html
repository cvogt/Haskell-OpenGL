   35 moveAllObjects :: Game t ()
   36 moveAllObjects = do
   37     w <- get
   38     let ents = worldEntities w
   39         newEnts = map (\o -> movePos o (Vec3 1 0 0)) ents
   40     put w{worldEntities = newEnts}

   42 -- | A class for types that have a position
   43 --   that can be retrieved and set.
   44 class HasPosition p where
   45     getPos :: p -> Vec3 GLfloat
   46     setPos :: p -> Vec3 GLfloat -> p
   47     movePos :: p -> Vec3 GLfloat -> p
   48     movePos hp movement =
   49         setPos hp (getPos hp + movement)
   50 
   51 -- | A class for types that have a rotation
   52 --   that can be retrieved and set.
   53 class HasRotation r where
   54     getRot :: r -> Vec3 GLfloat
   55     setRot :: r -> Vec3 GLfloat -> r
   56     rotate :: r -> Vec3 GLfloat -> r
   57     rotate r deltaR =
   58         setRot r (getRot r + deltaR)
   59 
   60 -- | A class for types that have an
   61 --   Axis-Aligned Bounding Box (AABB). Type must
   62 --   also have a position for this to make sense.
   63 class HasPosition a => HasAABB a where
   64     getAABBs :: a -> [AABB]
   65     getWholeAABB :: a -> Maybe AABB
   66 
   67 instance HasPosition (Vec3 GLfloat) where
   68     getPos = id
   69     setPos _ = id

   71 -- | The type used to contain the overall state of
   72 --   the entire game.
   73 data World t = World {
   74     worldPlayer :: !(GameObject t),
   75     worldEntities :: ![GameObject t],
   76     worldOctree :: !(Octree AABB),
   77     worldGraphics :: !(Graphics t),
   78     worldState :: !WorldState
   79 }

   81 -- | The type used to contain global values relating to
   82 --   graphics / shaders.
   83 data Graphics t = Graphics {
   84     graphicsUniforms :: ![ShaderUniform],
   85     graphicsPostProcessors :: !(Framebuffer, [GLuint]),
   86     graphicsShadowInfo :: !(Framebuffer, GLuint)

   89 emptyGraphics :: Graphics t
   90 emptyGraphics = Graphics [] (undefined, []) (undefined, 0)

  104 -- | Type used to generalize any update that:
  105 --   1. Involves a specific type 't'.

  112 -- | Type representing nearly anything in the game
  113 --   that is somewhat "physical".
  114 data GameObject t = Player {
  115     playerPosition :: !(Vec3 GLfloat),
  116     playerRotation :: !(Vec3 GLfloat),
  117     playerVelocity :: !(Vec3 GLfloat),
  118     playerSpeed :: !GLfloat,
  119     playerUpdate :: !(Game t (World t)),
  120     playerInput :: !(Input t)
  121 } | Entity {
  122     entityPosition :: !(Vec3 GLfloat),
  123     entityRotation :: !(Vec3 GLfloat),
  124     entityUpdate :: !(GameObject t -> Game t (GameObject t)),
  125     entityModel :: !Model,
  126     entityAttribute :: !t
  127 }
  128 
  129 emptyEntity :: GameObject ()
  130 emptyEntity = Entity 0 0 return emptyModel ()

  140 instance HasPosition (GameObject t) where
  141     getPos p@(Player{}) = playerPosition p
  142     getPos pe@(Entity{}) = entityPosition pe
  143 
  144     setPos p@(Player{}) pos = p{playerPosition = pos}
  145     setPos pe@(Entity{}) pos = pe{entityPosition = pos}
  146 
  147 instance HasRotation (GameObject t) where
  148     getRot p@(Player{}) = playerRotation p
  149     getRot e@(Entity{}) = entityRotation e
  150 
  151     setRot p@(Player{}) rot = p{playerRotation = rot}
  152     setRot e@(Entity{}) rot = e{entityRotation = rot}
  153 
  154 instance HasAABB (GameObject t) where
  155     getAABBs (Player{}) = [playerAABB]
  156     getAABBs pe@(Entity{}) =
  157         let aabbs = modelAABBs $ entityModel pe
  158         in fromMaybe [] aabbs
  159 
  160     getWholeAABB (Player{}) = Just playerAABB
  161     getWholeAABB pe@(Entity{}) =
  162         modelWholeAABB $ entityModel pe

  164 data Model = Model {
  165     modelShader :: !Shader,
  166     modelShaderVars :: ![ShaderAttrib],
  167     modelTextures :: ![Texture],
  168     modelVertCount :: !GLint,
  169     modelAABBs :: !(Maybe [AABB]),
  170     modelWholeAABB :: !(Maybe AABB)
  171 }
  172 
  173 emptyModel :: Model
  174 emptyModel = Model (Shader 0 []) [] [] 0 (Just [AABB 0 0]) (Just $ AABB 0 0)

  176 data Octree a where
  177     ONode :: HasAABB a => AABB -> [Octree a] -> Octree a
  178     OLeaf :: HasAABB a => AABB -> [a] -> Int -> Octree a
  179 
  180 instance Show a => Show (Octree a) where
  181     show (ONode aabb children) =
  182         "Node: " ++ show aabb ++ "\n{\n" ++
  183         concatMap ((++"\n") . show) children ++ "}"
  184     show (OLeaf aabb contents _) =
  185         "Leaf: " ++ show aabb ++ ", " ++ show contents

  198 -- | AABB (min corner) (max corner)
  199 data AABB = AABB !(Vec3 GLfloat) !(Vec3 GLfloat) deriving (Show)
  200 
  201 instance HasPosition AABB where
  202     getPos (AABB minV _) = minV
  203     setPos (AABB minV maxV) pos =
  204         AABB pos ((maxV - minV) + pos)
  205 
  206 instance HasAABB AABB where
  207     getWholeAABB (AABB low high) = Just (AABB 0 (high - low))
  208     getAABBs (AABB low high) = [AABB 0 (high - low)]
