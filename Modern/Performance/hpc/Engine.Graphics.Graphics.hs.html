   23 renderWorldMat :: World t -> IO (World t)
   24 renderWorldMat world = do
   25     glClear $ gl_COLOR_BUFFER_BIT .|. gl_DEPTH_BUFFER_BIT
   26 
   27     -- Get window dimensions from GLFW
   28     dimensions <- GLFW.getWindowSize
   29                 (fromJust $ windowInner $ stateWindow $ worldState world)
   30 
   31     -- Update matrices.
   32     let worldMats = calculateMatricesFromPlayer
   33                         (worldPlayer world) dimensions
   34 
   35     -- Render world with matrices.
   36     newEntites <- renderObjectsMat world worldMats (worldEntities world)
   37     return world{worldEntities = newEntites}
   38 
   39 renderObjectsMat :: World t -> WorldMatrices -> [GameObject t] -> IO [GameObject t]
   40 renderObjectsMat world wm (object:rest) = do
   41     let model = getModel object
   42         Vec3 objx objy objz = getPos object
   43         Vec3 objrx objry objrz = getRot object
   44         mShader = modelShader model
   45 
   46         -- Move Object
   47         modelMat = gtranslationMatrix [objx, objy, objz] *
   48                    grotationMatrix [objrx, objry, objrz]
   49 
   50     -- Use object's shader
   51     glUseProgram $ shaderId mShader
   52 
   53     -- Set uniforms. (World uniforms and Matrices).
   54     newShader <-
   55         setMatrixUniforms mShader wm{matrixModel = modelMat}
   56 
   57     -- Bind buffers to variable names in shader.
   58     setShaderAttribs $ modelShaderVars model
   59     bindTextures (modelTextures model) $ shaderId newShader
   60 
   61     -- Do the drawing.
   62     glDrawArrays gl_TRIANGLES 0 (modelVertCount model)
   63 
   64     -- TODO: Remove if not necessary.
   65     -- Disable textures.
   66     --unBindTextures (fromIntegral . length . modelTextures $ model)
   67 
   68     -- Turn off VBO/VAO
   69     disableShaderAttribs $ modelShaderVars model
   70 
   71     -- Disable the object's shader.
   72     glUseProgram 0
   73 
   74     -- Update the object's shader
   75     let newObject = object{entityModel =
   76                     (entityModel object){modelShader = newShader}}
   77 
   78     restObjects <- renderObjectsMat world wm rest
   79 
   80     return $ newObject : restObjects
   81 renderObjectsMat _ _ [] = return []

   87 initGL :: GLFW.Window -> IO ()
   88 initGL win = do
   89     -- Enables smooth color shading.
   90     --glShadeModel gl_SMOOTH
   91 
   92     -- Set "background color" to black
   93     glClearColor 0 0 0 0
   94 
   95     -- Enables clearing of the depth buffer
   96     glClearDepth 1
   97     -- Allow depth testing (3D)
   98     glEnable gl_DEPTH_TEST
   99     -- Tells OpenGL how to deal with overlapping shapes
  100     glDepthFunc gl_LESS
  101     --glDepthFunc gl_LEQUAL
  102 
  103     -- Tell OpenGL to use the nicest perspective correction.
  104     -- The other choices are gl_FASTEST and gl_DONT_CARE.
  105     glHint gl_PERSPECTIVE_CORRECTION_HINT gl_NICEST
  106 
  107     -- Enable culling of faces.
  108     glEnable gl_CULL_FACE
  109     -- Do not render the backs of faces. Increases performance.
  110     glCullFace gl_BACK
  111 
  112     -- Enable textures.
  113     --glEnable gl_TEXTURE
  114     glTexParameteri gl_TEXTURE_2D gl_TEXTURE_MAG_FILTER
  115         (fromIntegral gl_LINEAR)
  116     glTexParameteri gl_TEXTURE_2D gl_TEXTURE_MIN_FILTER
  117         (fromIntegral gl_LINEAR_MIPMAP_LINEAR)
  118     glTexParameteri gl_TEXTURE_2D gl_TEXTURE_WRAP_S
  119         (fromIntegral gl_CLAMP_TO_EDGE)
  120     glTexParameteri gl_TEXTURE_2D gl_TEXTURE_WRAP_T
  121         (fromIntegral gl_CLAMP_TO_EDGE)
  122 
  123     -- Call resize function.
  124     (w,h) <- GLFW.getFramebufferSize win
  125     resizeScene win w h 
  126 
  127 resizeScene :: GLFW.WindowSizeCallback
  128 -- Prevent divide by 0
  129 resizeScene win w 0 = resizeScene win w 1
  130 resizeScene _ width height =
  131     -- Make viewport the same size as the window.
  132     glViewport 0 0 (fromIntegral width) (fromIntegral height)

  134 cleanupWorld :: World t -> IO ()
  135 cleanupWorld world = do
  136     let fb = fst $ graphicsPostProcessors $ worldGraphics world
  137         shaders = snd $ graphicsPostProcessors $ worldGraphics world
  138     with (fbufName fb) $ glDeleteFramebuffers 1
  139     with (fbufTexture fb) $ glDeleteTextures 1
  140     with (fbufVBO fb) $ glDeleteVertexArrays 1
  141     mapM_ glDeleteProgram shaders
  142     with (fbufRenderBuffer fb) $ glDeleteRenderbuffers 1
  143 
  144 cleanupObjects :: [GameObject t] -> IO ()
  145 cleanupObjects (object:rest) = do
  146     -- Delete buffers.
  147     let shaderVarAttrIds = map (\(Vec3 attrId _ _) -> attrId)
  148                               (modelShaderVars $ getModel object)
  149         shaderVarBufIds = map (\(Vec3 _ bufId _) -> bufId)
  150                               (modelShaderVars $ getModel object)
  151     mapM_ (\x -> with x $ glDeleteBuffers 1) shaderVarBufIds
  152 
  153     -- Delete shader.
  154     glDeleteProgram (shaderId $ modelShader $ getModel object)
  155 
  156     -- Delete textures.
  157     let model = getModel object
  158         textures = map (\(GL.TextureObject tid, _) -> tid) $ modelTextures model
  159     mapM_ (\x -> with x $ glDeleteTextures 1) textures
  160 
  161     -- Delete vertex arrays.
  162     mapM_ (\x -> with x $ glDeleteVertexArrays 1) shaderVarAttrIds
  163 
  164     print "cleanup"
  165 
  166     cleanupObjects rest
  167 cleanupObjects [] = return ()
