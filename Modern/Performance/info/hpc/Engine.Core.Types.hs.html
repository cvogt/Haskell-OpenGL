   64 -- | A class for types that have a position
   65 --   that can be retrieved and set.
   66 class HasPosition p where
   67     getPos :: p -> Vec3 GLfloat
   68     setPos :: p -> Vec3 GLfloat -> p
   69     movePos :: p -> Vec3 GLfloat -> p
   70     movePos hp movement =
   71         setPos hp (getPos hp + movement)
   72 
   73 -- | A class for types that have a rotation
   74 --   that can be retrieved and set.
   75 class HasRotation r where
   76     getRot :: r -> Vec3 GLfloat
   77     setRot :: r -> Vec3 GLfloat -> r
   78     rotate :: r -> Vec3 GLfloat -> r
   79     rotate r deltaR =
   80         setRot r (getRot r + deltaR)

  167 instance HasPosition (GameObject t) where
  168     getPos p@(Player{}) = playerPosition p
  169     getPos pe@(Entity{}) = entityPosition pe
  170 
  171     setPos p@(Player{}) pos = p{playerPosition = pos}
  172     setPos pe@(Entity{}) pos = pe{entityPosition = pos}
  173 
  174 instance HasRotation (GameObject t) where
  175     getRot p@(Player{}) = playerRotation p
  176     getRot e@(Entity{}) = entityRotation e
  177 
  178     setRot p@(Player{}) rot = p{playerRotation = rot}
  179     setRot e@(Entity{}) rot = e{entityRotation = rot}
  180 
  181 instance HasAABB (GameObject t) where
  182     getAABBs (Player{}) = [playerAABB]
  183     getAABBs pe@(Entity{}) =
  184         let aabbs = modelAABBs $ entityModel pe
  185         in fromMaybe [] aabbs
  186 
  187     getWholeAABB (Player{}) = Just playerAABB
  188     getWholeAABB pe@(Entity{}) =
  189         modelWholeAABB $ entityModel pe

  208 -- | AABB (min corner) (max corner)
  209 data AABB = AABB (Vec3 GLfloat) (Vec3 GLfloat) deriving (Show)
  210 
  211 instance HasPosition AABB where
  212     getPos (AABB minV _) = minV
  213     setPos (AABB minV maxV) pos =
  214         AABB pos ((maxV - minV) + pos)
  215 
  216 instance HasAABB AABB where
  217     getWholeAABB (AABB low high) = Just (AABB 0 (high - low))
  218     getAABBs (AABB low high) = [AABB 0 (high - low)]

  220 -- | A pure Octree used to sort objects for
  221 --   collision detection.
  222 data Octree a =
  223     ONode AABB [Octree a]
  224   | OLeaf AABB [a] Int
  225 
  226 data MOctree a =
  227     MNode AABB (MV.IOVector (MOctree a))
  228   | PNode AABB (V.Vector (MOctree a))
  229   | MLeaf AABB (MV.IOVector a) Int
  230   | PLeaf AABB (V.Vector a) Int
  231 
  232 instance Show a => Show (Octree a) where
  233     show (ONode aabb children) =
  234         "Node: " ++ show aabb ++ "\n{\n" ++
  235         concatMap ((++"\n") . show) children ++ "}"
  236     show (OLeaf aabb contents _) =
  237         "Leaf: " ++ show aabb ++ ", " ++ show contents

  333 -- | 4x4 Matrix in the OpenGL orientation:
  334 --   translation column is the last 4 elements.
  335 type Matrix4x4 = [[GLfloat]]
  336 -- | 3x3 Matrix in the OpenGL orientation.
  337 type Matrix3x3 = [[GLfloat]]
  338 -- | Four element GLfloat vector.
  339 type Vector4 = [GLfloat]
  340 -- | Three element GLfloat vector.
  341 type Vector3 = [GLfloat]
  342 
  343 instance Num Matrix4x4 where
  344     a * b =
  345         map (\row -> map (gdotVec row) at) b
  346         where at = transpose a
  347     a + b = applyToIndices2 a b (+)
  348     abs = map (map abs)
  349     fromInteger i =
  350         [
  351         [fromInteger i, 0, 0, 0],
  352         [0, fromInteger i, 0, 0],
  353         [0, 0, fromInteger i, 0],
  354         [0, 0, 0, fromInteger i]
  355         ]
  356     signum = map $ map signum
  357     negate = map $ map negate
  358 
  359 applyToIndices2 :: [[a]] -> [[b]] -> (a -> b -> c) -> [[c]]
  360 applyToIndices2 (a:as) (b:bs) f =
  361     applyToIndices a b f : applyToIndices2 as bs f
  362 applyToIndices2 _ _ _ = []
  363 applyToIndices :: [a] -> [b] -> (a -> b -> c) -> [c]
  364 applyToIndices (a:as) (b:bs) f =
  365     f a b : applyToIndices as bs f
  366 applyToIndices _ _ _ = []
  367 -- | Vector dot product.
  368 gdotVec :: [GLfloat] -> [GLfloat] -> GLfloat
  369 gdotVec a b = sum $ ginnerVec a b
  370 -- | Inner product of two vectors.
  371 ginnerVec :: [GLfloat] -> [GLfloat] -> [GLfloat]
  372 ginnerVec = zipWith (*)
