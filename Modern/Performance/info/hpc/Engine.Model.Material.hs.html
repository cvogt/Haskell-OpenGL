   21 data Material = Material {
   22     matName :: B.ByteString,
   23     matAmbientColor :: Maybe (Vec3 GLfloat),
   24     matDiffuseColor :: Maybe (Vec3 GLfloat),
   25     matSpecularColor :: Maybe (Vec3 GLfloat),
   26     matTexture :: Maybe GLuint,
   27     matTexId :: Maybe GLint,
   28     matTexturePaths :: [B.ByteString]
   29 } deriving (Show)
   30 
   31 instance NFData Material where
   32     rnf (Material n ac dc sc t ti tps) =
   33         n `deepseq` ac `deepseq` dc
   34         `deepseq` sc `deepseq` t `seq` ti `seq` tps `deepseq` ()
   35 
   36 loadMtlFile :: FilePath -> IO [Material]
   37 loadMtlFile file = do
   38     let directory = (intercalate "/" . init $ splitOn "/" file) ++ "/"
   39     openFile file ReadMode >>= loadMtlMaterials directory
   40 
   41 
   42 loadMtlMaterials :: FilePath -> Handle -> IO [Material]
   43 loadMtlMaterials directory handle =
   44     liftM (map applyDefualtMtl . tail)
   45           (loadMtlMaterialsRec directory 0 handle emptyMaterial)
   46 
   47 -- | Apply defualt values to attributes set to
   48 --   Nothing according to spec at
   49 --   http://people.sc.fsu.edu/~jburkardt/data/mtl/mtl.html
   50 applyDefualtMtl :: Material -> Material
   51 applyDefualtMtl mat@(Material _ amb diff spec _ texId _) =
   52     let newAmb = if isNothing amb
   53                     then Just $ Vec3 0.2 0.2 0.2
   54                 else amb
   55         newDiff = if isNothing diff
   56                     then Just $ Vec3 0.8 0.8 0.8
   57                 else diff
   58         newSpec = if isNothing spec
   59                     then Just $ Vec3 1.0 1.0 1.0
   60                 else spec
   61         newTexId = if isNothing texId
   62                     then Just (-1)
   63                 else texId
   64     in mat{matAmbientColor = newAmb,
   65            matDiffuseColor = newDiff,
   66            matSpecularColor = newSpec,
   67            matTexId = newTexId}
   68 
   69 -- | UNSAFE!! Use loadMtlMaterials instead.
   70 loadMtlMaterialsRec :: FilePath -> GLuint -> Handle -> Material -> IO [Material]
   71 loadMtlMaterialsRec directory textureCount handle start = do
   72     eof <- hIsEOF handle
   73     if not eof
   74         then do 
   75             line <- B.hGetLine handle
   76             -- If there is a declaration of a new Material,
   77             -- "add the current mat to the list" and start
   78             -- on a new Material.
   79             if "newmtl " `B.isPrefixOf` line
   80                 then do
   81                     let name = head $ rawMtlLine line
   82                     rest <- loadMtlMaterialsRec directory textureCount handle
   83                                 (Material name Nothing Nothing Nothing Nothing Nothing [])
   84                     return $ start : rest
   85             else if not $ B.null line
   86                 -- Call executeCommand on current line and Material
   87                 -- and then continue adding attributes to that Material.
   88                 then
   89                     let origTex = matTexture start
   90                     in do
   91                         newMat <- executeCommand directory line start textureCount
   92                         if matTexture newMat == origTex
   93                             then loadMtlMaterialsRec directory textureCount handle newMat
   94                         else loadMtlMaterialsRec directory (textureCount+1) handle newMat
   95             else loadMtlMaterialsRec directory textureCount handle start
   96     -- If it is End Of File, close the file and
   97     -- return the last Material.
   98     else hClose handle >> return [start]
   99 
  100 executeCommand :: FilePath -> B.ByteString -> Material -> GLuint -> IO Material
  101 executeCommand directory command mat textureCount
  102     | "Ka " `B.isPrefixOf` command =
  103         return mat{matAmbientColor = Just $ readMtlLineTriplet command}
  104     | "Kd " `B.isPrefixOf` command =
  105         return mat{matDiffuseColor = Just $ readMtlLineTriplet command}
  106     | "Ks " `B.isPrefixOf` command =
  107         return mat{matSpecularColor = Just $ readMtlLineTriplet command}
  108     | "map_Kd " `B.isPrefixOf` command = do
  109         texture <- juicyLoadTexture $ directory ++ B.unpack (head $ rawMtlLine command)
  110         return mat{matTexture = Just texture,
  111                    matTexId = Just $ fromIntegral textureCount,
  112                    matTexturePaths = head (rawMtlLine command) : matTexturePaths mat}
  113     | otherwise = return mat

  121 readMtlLineTriplet :: B.ByteString -> Vec3 GLfloat
  122 readMtlLineTriplet = toTripletMtl . readMtlLine
  123 
  124 readMtlLine :: B.ByteString -> [GLfloat]
  125 readMtlLine = map parseBsFloat . tail . filter (not . B.null) . B.split ' '
  126 
  127 rawMtlLine :: B.ByteString -> [B.ByteString]
  128 rawMtlLine = tail . filter (not . B.null) . B.split ' '
  129 
  130 toTripletMtl :: [a] -> Vec3 a
  131 toTripletMtl xs
  132     | length xs == 3 = Vec3 (head xs) (xs !! 1) (xs !! 2)
  133     | otherwise = error "Material.toTripletMtl"
  134 
  135 emptyMaterial :: Material
  136 emptyMaterial = Material "" Nothing Nothing Nothing Nothing Nothing []
  137 
  138 parseBsFloat :: B.ByteString -> GLfloat
  139 parseBsFloat = fst . parseBsFloat'
  140 
  141 parseBsFloat' :: B.ByteString -> (GLfloat, B.ByteString)
  142 parseBsFloat' bs
  143     | isJust ('e' `B.elemIndex` bs) =
  144         let [baseBs, expBs] = B.splitWith (=='e') bs
  145             (base, _) = parseBsFloat' baseBs
  146             (expon, rest) = parseBsFloat' expBs
  147         in (base * (10.0 ** expon), rest)
  148 
  149     | isJust ('.' `B.elemIndex` bs) =
  150         let negative = B.head bs == '-'
  151             Just (whole, decimalBS) = B.readInt $
  152                 if negative
  153                     then B.tail bs
  154                 else bs
  155             Just (decimal, rest) = B.readInt $ B.tail decimalBS
  156         in if negative
  157                 then (negate $ fromIntegral whole +
  158                     (fromIntegral decimal / 10 ^ (B.length decimalBS - 1)), rest)
  159             else (fromIntegral whole +
  160                 (fromIntegral decimal / 10 ^ (B.length decimalBS - 1)), rest)
  161 
  162     | otherwise =
  163         let Just (val, rest) = B.readInt bs
  164         in (fromIntegral val, rest)
