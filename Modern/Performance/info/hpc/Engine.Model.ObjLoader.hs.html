   34 loadObjModel ::
   35     FilePath ->
   36     FilePath ->
   37     FilePath ->
   38     IO Model
   39 loadObjModel objFile vert frag =
   40     let attrNames = ["position", "texCoord", "normal", "color", "textureId"]
   41     in do
   42         hasDatFile <- doesFileExist $ objFile ++ ".dat"
   43         if hasDatFile
   44             then loadDatModel (objFile ++ ".dat") vert frag
   45         else do
   46             fileContents <- getContentsSafe objFile
   47             let directory = directoryOfFile objFile
   48 
   49             (mats, lib) <- loadObjMaterials directory fileContents
   50 
   51             let total = totalObjData fileContents mats
   52                 images = concatMap matTexturePaths lib
   53 
   54             writeDataToFile (objFile ++ ".dat") total images
   55 
   56             let (totalData, mTextures) =
   57                     (total, map (fromJust . matTexture) $ filter (isJust . matTexture) lib)
   58 
   59             tmp <- createModel vert frag
   60                 attrNames
   61                 totalData
   62                 [3, 2, 3, 3, 1]
   63                 (fromIntegral (length . head $ totalData) `div` 3)
   64 
   65             let mTexIds = replicate (length mTextures) 0 :: [GLint]
   66             return tmp{modelTextures =
   67                 zip mTextures
   68                     mTexIds}

   70 totalObjData :: B.ByteString -> [Material] -> [[GLfloat]]
   71 totalObjData contents mats =
   72     let obj = loadObj contents
   73         dat = toArrays obj
   74         materialDiffs = fromVec3M $ map matDiffuseColor mats
   75         materialTexIds = map (fromIntegral . fromJustSafe . matTexId) mats
   76     in dat ++ [materialDiffs, materialTexIds]

   78 getContentsSafe :: FilePath -> IO B.ByteString
   79 getContentsSafe f = do
   80     handle <- openFile f ReadMode
   81     fileContents <- B.hGetContents handle
   82     fileContents `deepseq` hClose handle
   83     return fileContents
   84 
   85 directoryOfFile :: FilePath -> FilePath
   86 directoryOfFile = (++"/") . intercalate "/" . init . splitOn "/"

   88 -- | Parse an .obj file and return a Vec3 [GLfloat]
   89 --   containing the vertices, texture coordinates,
   90 --   and normals, in that order.
   91 loadObjFile :: FilePath -> IO (Vec3 [GLfloat])
   92 loadObjFile file = do
   93     handle <- openFile file ReadMode
   94     objData <- liftM loadObj $! B.hGetContents handle
   95     hClose handle
   96     return objData
   97 
   98 -- | Parse a string containing the specification
   99 --   for an .obj model and return a Vec3 [GLfloat]
  100 --   containing the vertices, texture coordinates,
  101 --   and normals, in that order.
  102 loadObj :: B.ByteString -> Vec3 [GLfloat]
  103 loadObj text =
  104     let verts = parseVertices text
  105         norms = parseNormals text
  106         texs = parseTextures text
  107         faces = parseFaces text
  108     in packObj verts norms texs faces
  109 
  110 -- | Given DLists containg raw information read from a
  111 --   .obj file, put together a Vec3 [GLfloat] of the
  112 --   vertices, texture coordinates, and normals,
  113 --   in that order, formatted to be sent to OpenGL.
  114 packObj ::
  115     D.DList GLfloat ->      -- ^ Vertices
  116     D.DList GLfloat ->      -- ^ Normals
  117     D.DList GLfloat ->      -- ^ Texture coords
  118     D.DList (Maybe Int) ->  -- ^ Face Definitions
  119     Vec3 [GLfloat]
  120 packObj verts norms texs faces =
  121     let lFaces = D.toList faces
  122         lVerts = V.fromList $ D.toList verts
  123         lNorms = V.fromList $ D.toList norms
  124         lTexs = V.fromList $ D.toList texs
  125 
  126         faceVerts = takeEvery3 1 lFaces
  127         faceTexs = takeEvery3 2 lFaces
  128         faceNorms = takeEvery3 3 lFaces
  129 
  130         realVerts = D.toList $ getIndicesD lVerts faceVerts 3
  131         realNorms = D.toList $ getIndicesD lNorms faceNorms 3
  132         realTexs = D.toList $ getIndicesD lTexs faceTexs 2
  133     in Vec3 realVerts realTexs realNorms
  134 
  135 -- | Given a List of () indices to read and the
  136 --   Vector to read from, read given indices for all
  137 --   Just values, and use -1.0 as a default value for
  138 --   Nothings. Returns a DList of GLfloats.
  139 getIndicesD :: V.Vector GLfloat -> [Maybe Int] -> Int -> D.DList GLfloat
  140 getIndicesD haystack (x:xs) vecType
  141     | isJust x =
  142             -- Get the section of the vector that is needed.
  143         let index = vecType * (fromJust x - 1)
  144             splitList = snd $ V.splitAt index haystack
  145             -- Switch to dlist for its O(1) appending.
  146             ret = D.fromList $ V.toList $ V.take vecType splitList
  147         in ret `D.append` getIndicesD haystack xs vecType
  148     | otherwise = D.replicate vecType (-1.0) `D.append` getIndicesD haystack xs vecType
  149 getIndicesD _ _ _ = D.empty
  150 
  151 takeEvery3 :: Int -> [a] -> [a]
  152 takeEvery3 offset (x:y:z:rest)
  153     | offset == 1 = x : takeEvery3 offset rest
  154     | offset == 2 = y : takeEvery3 offset rest
  155     | offset == 3 = z : takeEvery3 offset rest
  156     | otherwise = error $ "ObjLoader.takeEvery 3: arg"
  157                         ++ "must be 1, 2, or 3"
  158 takeEvery3 _ _ = []
  159 {-# INLINE takeEvery3 #-}
  160 
  161 parseFaces :: B.ByteString -> D.DList (Maybe Int)
  162 parseFaces = D.concat . map faceLineCheck . B.lines
  163     where
  164     faceLineCheck line =
  165         if "f " `B.isPrefixOf` line
  166             then parseFaceLine line
  167         else D.empty
  168 
  169 parseFaceLine :: B.ByteString -> D.DList (Maybe Int)
  170 parseFaceLine =
  171     D.concat . map parseFaceGroup . tail . B.words
  172 {-# INLINE parseFaceLine #-}
  173 
  174 parseFaceGroup :: B.ByteString -> D.DList (Maybe Int)
  175 parseFaceGroup =
  176     D.fromList . map retrieveData . B.split '/'
  177     where
  178     retrieveData text =
  179         if B.null text
  180             then Nothing
  181         else Just $ (floor . parseBsFloat) text
  182 {-# INLINE parseFaceGroup #-}
  183 
  184 parseVertices :: B.ByteString -> D.DList GLfloat
  185 parseVertices = parsePrefix "v "
  186 parseNormals :: B.ByteString -> D.DList GLfloat
  187 parseNormals = parsePrefix "vn "
  188 parseTextures :: B.ByteString -> D.DList GLfloat
  189 parseTextures =
  190     D.concat . map lineCheck . B.lines
  191     where
  192     lineCheck :: B.ByteString -> D.DList GLfloat
  193     lineCheck line =
  194         if "vt " `B.isPrefixOf` line
  195             then
  196                 let [x, y] = D.toList $ parseLine line
  197                 in D.fromList [x, 1-y]
  198         else D.empty
  199 
  200 -- | Call parseLine on all lines with the given prefix
  201 --   in given String.
  202 parsePrefix :: B.ByteString -> B.ByteString -> D.DList GLfloat
  203 parsePrefix prefix =
  204     D.concat . map lineCheck . B.lines
  205     where
  206     lineCheck line =
  207         if prefix `B.isPrefixOf` line
  208             then parseLine line
  209         else D.empty
  210     {-# INLINE lineCheck #-}
  211 
  212 parseLine :: B.ByteString -> D.DList GLfloat
  213 parseLine = D.map parseBsFloat . D.tail . D.fromList . B.words
  214 {-# INLINE parseLine #-}

  216 loadObjMaterials :: String -> B.ByteString -> IO ([Material], [Material])
  217 loadObjMaterials directory contents = do
  218     library <- loadObjMaterialLib directory contents
  219 
  220     let listRet = listOfMats contents library emptyMaterial
  221 
  222     return (listRet, library)
  223 
  224 loadObjMaterialLib :: String -> B.ByteString -> IO [Material]
  225 loadObjMaterialLib directory =
  226     liftM concat . mapM (\x -> loadMtlFile (directory++B.unpack x)) . parseMtlLibs
  227 {-# INLINE loadObjMaterialLib #-}
  228 
  229 parseMtlLibs :: B.ByteString -> [B.ByteString]
  230 parseMtlLibs = getMaterialLibs . B.lines
  231 
  232 getMaterialLibs :: [B.ByteString] -> [B.ByteString]
  233 getMaterialLibs (line:rest) =
  234     if "mtllib " `B.isPrefixOf` line
  235         then (last . B.words $ line) :
  236                 getMaterialLibs rest
  237     else getMaterialLibs rest
  238 getMaterialLibs _ = []
  239 
  240 listOfMats :: B.ByteString -> [Material] -> Material -> [Material]
  241 listOfMats contents =
  242     lineAction (B.lines contents)
  243     where
  244     lineAction :: [B.ByteString] -> [Material] -> Material -> [Material]
  245     lineAction (line:rest) library currentMat
  246         | "usemtl " `B.isPrefixOf` line =
  247             let mat = findMaterial (last . B.words $ line) library
  248             in lineAction rest library mat
  249         | "f " `B.isPrefixOf` line =
  250             replicate 3 currentMat ++
  251                 lineAction rest library currentMat
  252         | otherwise = lineAction rest library currentMat
  253     lineAction _ _ _ = []
  254 
  255 findMaterial :: B.ByteString -> [Material] -> Material
  256 findMaterial name library = head $ filter (\x -> matName x == name) library
  257 {-# INLINE findMaterial #-}

  259 toArrays :: Vec3 [a] -> [[a]]
  260 toArrays (Vec3 x y z) = [x, y, z]
  261 {-# INLINE toArrays #-}
  262 
  263 fromVec3M :: [Maybe (Vec3 a)] -> [a]
  264 fromVec3M (Just (Vec3 x y z) : xs) =
  265     [x, y, z] ++ fromVec3M xs
  266 fromVec3M [] = []
  267 fromVec3M (Nothing : _) = error "fromVec3M: argument contained Nothing."

  269 fromJustSafe :: Num a => Maybe a -> a
  270 fromJustSafe (Just x) = x
  271 fromJustSafe Nothing = 0
  272 {-# INLINE fromJustSafe #-}

  274 parseBsFloat :: B.ByteString -> GLfloat
  275 parseBsFloat = fst . parseBsFloat'
  276 
  277 parseBsFloat' :: B.ByteString -> (GLfloat, B.ByteString)
  278 parseBsFloat' bs
  279     | isJust ('e' `B.elemIndex` bs) =
  280         let [baseBs, expBs] = B.splitWith (=='e') bs
  281             (base, _) = parseBsFloat' baseBs
  282             (expon, rest) = parseBsFloat' expBs
  283         in (base * (10.0 ** expon), rest)
  284 
  285     | isJust ('.' `B.elemIndex` bs) =
  286         let negative = B.head bs == '-'
  287             Just (whole, decimalBS) = B.readInt $
  288                 if negative
  289                     then B.tail bs
  290                 else bs
  291             Just (decimal, rest) = B.readInt $ B.tail decimalBS
  292         in if negative
  293                 then (negate $ fromIntegral whole +
  294                     (fromIntegral decimal / 10 ^ (B.length decimalBS - 1)), rest)
  295             else (fromIntegral whole +
  296                 (fromIntegral decimal / 10 ^ (B.length decimalBS - 1)), rest)
  297 
  298     | otherwise =
  299         let Just (val, rest) = B.readInt bs
  300         in (fromIntegral val, rest)
