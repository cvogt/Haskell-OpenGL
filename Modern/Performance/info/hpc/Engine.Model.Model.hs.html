1 module Engine.Model.Model (
    2     Model(..), createModel, createModelWithProgram
    3 ) where
    4 
    5 import Graphics.Rendering.OpenGL.Raw
    6     (GLfloat, GLuint, GLint)
    7 
    8 import Engine.Core.Types
    9     (Model(..), Shader(..))
   10 import Engine.Graphics.Shaders
   11     (loadProgram, getAttrLocs, createShaderAttribs)
   12 import Engine.Model.AABB (aabbByFace, aabbFromPoints)
   13 import Engine.Graphics.GraphicsUtils (createBufferIdAll)
   14 
   15 createModel ::
   16     FilePath ->     -- ^ Vertex Shader.
   17     FilePath ->     -- ^ Fragment Shader.
   18     [String] ->     -- ^ Attribute Variable names.
   19     [[GLfloat]] ->  -- ^ List containing all the lists of values.
   20                    --   (vertices, normals, etc).
   21     [GLuint] ->     -- ^ Size of each value.
   22     GLint ->        -- ^ Number of vertices.
   23     IO Model
   24 createModel vert frag attrNames buffData valLens vertCount = do
   25     program <- loadProgram vert frag
   26     attribs <- getAttrLocs program attrNames
   27     ids <- createBufferIdAll buffData
   28 
   29     let sAttribs = createShaderAttribs attribs ids valLens
   30     return $ Model (Shader program []) sAttribs [] vertCount
   31             (Just $ aabbByFace (head buffData))
   32             (Just $ aabbFromPoints (head buffData))
   33
