   21 -- | Load an image and turn it into something OpenGL can use.
   22 juicyLoadTexture :: FilePath -> IO GLuint
   23 juicyLoadTexture file = do
   24     (Image (w, h) pd) <- juicyLoadImageRaw file
   25     texName <- alloca $ \buf -> do
   26         glGenTextures 1 buf
   27         peek buf
   28 
   29     glBindTexture gl_TEXTURE_2D texName
   30 
   31     glTexParameteri gl_TEXTURE_2D
   32         gl_TEXTURE_MIN_FILTER (fromIntegral gl_NEAREST)
   33     glTexParameteri gl_TEXTURE_2D
   34         gl_TEXTURE_MAG_FILTER (fromIntegral gl_NEAREST)
   35 
   36     glTexImage2D gl_TEXTURE_2D 0 (fromIntegral gl_RGB)
   37                  w h 0 (fromIntegral gl_RGB) gl_UNSIGNED_BYTE pd
   38 
   39     return texName
   40 
   41 -- TODO: add support for all (most) colorspaces / formats.
   42 juicyLoadImageRaw :: FilePath -> IO Image
   43 juicyLoadImageRaw file = do
   44     image <- Juicy.readImage file
   45 
   46     case image of
   47         Left err -> error err
   48 
   49         Right (Juicy.ImageRGB8 (Juicy.Image w h dat)) ->
   50             V.unsafeWith dat $ \ptr ->
   51             return $ Image (fromIntegral w, fromIntegral h) ptr
   52         Right (Juicy.ImageYCbCr8 img) ->
   53             let (Juicy.Image w h dat) =
   54                     JTypes.convertImage img :: Juicy.Image Juicy.PixelRGB8
   55             in V.unsafeWith dat $ \ptr ->
   56                 return $ Image (fromIntegral w, fromIntegral h) ptr
   57         Right (Juicy.ImageCMYK8 img) ->
   58             let (Juicy.Image w h dat) =
   59                     JTypes.convertImage img :: Juicy.Image Juicy.PixelRGB8
   60             in V.unsafeWith dat $ \ptr ->
   61                 return $ Image (fromIntegral w, fromIntegral h) ptr 
   62         _ -> error $
   63             "Engine.Graphics.Texture.juicyLoadImage:"
   64                 ++ "bad image colorspace or format."
