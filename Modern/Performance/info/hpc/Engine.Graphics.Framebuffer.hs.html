   41 -- | Render world with all postprocessing shaders defined by
   42 --   worldFramebuffer.
   43 renderWorldWithPostprocessing :: World t -> IO (World t)
   44 renderWorldWithPostprocessing world = do
   45     let effects = snd $ graphicsPostProcessors $ worldGraphics world 
   46     glBindFramebuffer gl_FRAMEBUFFER $
   47         fbufName $ fst $ graphicsPostProcessors $ worldGraphics world 
   48     ret <- renderWorldMat world
   49 
   50     renderAllPasses ret effects
   51     return ret
   52 
   53 renderAllPasses :: World t -> [GLuint] -> IO ()
   54 renderAllPasses world (shader:otherShader:rest) = do
   55     -- Render to FB.
   56     renderPostPass (fst $ graphicsPostProcessors $ worldGraphics world)
   57                    (worldState world) shader
   58 
   59     -- Make framebuffer use the new rendered image.
   60     glTexParameteri gl_TEXTURE_2D (fromIntegral gl_TEXTURE_MAG_FILTER)
   61                                   (fromIntegral gl_NEAREST)
   62     glTexParameteri gl_TEXTURE_2D (fromIntegral gl_TEXTURE_MIN_FILTER)
   63                                   (fromIntegral gl_NEAREST)
   64 
   65     -- Continue with other shaders.
   66     renderAllPasses world $ otherShader:rest
   67 
   68 renderAllPasses world (shader:[]) =
   69     let fb = fst $ graphicsPostProcessors $ worldGraphics world 
   70     -- If this is the last postprocessing shader,
   71     -- render it to the screen.
   72     in do
   73         glBindFramebuffer gl_FRAMEBUFFER 0
   74         uncurry (glViewport 0 0) $ fbufDimensions fb
   75         glClear $ gl_COLOR_BUFFER_BIT .|. gl_DEPTH_BUFFER_BIT
   76         renderPostPass fb (worldState world) shader
   77 renderAllPasses _ [] = return ()
   78 
   79 renderPostPass :: Framebuffer -> WorldState -> GLuint -> IO ()
   80 renderPostPass fb wState shader = do
   81     let quadVB = fbufVBO fb
   82 
   83     -- Activate the shader
   84     glUseProgram shader
   85 
   86     glActiveTexture gl_TEXTURE0
   87     glBindTexture gl_TEXTURE_2D $ fbufTexture fb
   88     texId <- withCString "renderedTexture" $ glGetUniformLocation shader
   89     glUniform1i texId 0
   90 
   91     let utcTime = stateTime wState
   92         dayTime = realToFrac $ utctDayTime utcTime
   93 
   94     setUniforms shader [("time", return [dayTime])]
   95 
   96     -- Enable the attribute buffer.
   97     glEnableVertexAttribArray 0
   98     -- Give OpenGL the information.
   99     glBindBuffer gl_ARRAY_BUFFER quadVB
  100     -- Tell OpenGL about the info.
  101     glVertexAttribPointer 0 3 gl_FLOAT 0 0 offset0
  102 
  103     glDrawArrays gl_TRIANGLES 0 6
  104 
  105     glDisableVertexAttribArray 0

  107 makeFrameBuffer :: (GLint, GLint) -> IO Framebuffer
  108 makeFrameBuffer (winW, winH) = do
  109     -- Create framebuffer and bind it.
  110     fbName <- alloca (\p -> glGenFramebuffers 1 p >> peek p)
  111     glBindFramebuffer gl_FRAMEBUFFER fbName
  112 
  113     -- Create a texture id.
  114     fbTexPtr <- new 0
  115     glGenTextures 1 fbTexPtr
  116     fbTex <- peek fbTexPtr
  117     glBindTexture gl_TEXTURE_2D fbTex
  118 
  119     -- Create an image.
  120     glTexImage2D gl_TEXTURE_2D 0
  121         (fromIntegral gl_RGB)
  122         winW winH
  123         0 gl_RGB gl_UNSIGNED_BYTE offset0
  124 
  125     -- Give texture paramenters.
  126     glTexParameteri gl_TEXTURE_2D (fromIntegral gl_TEXTURE_MAG_FILTER)
  127                                   (fromIntegral gl_NEAREST)
  128     glTexParameteri gl_TEXTURE_2D (fromIntegral gl_TEXTURE_MIN_FILTER)
  129                                   (fromIntegral gl_NEAREST)
  130     glTexParameteri gl_TEXTURE_2D gl_TEXTURE_WRAP_S
  131         $ fromIntegral gl_CLAMP_TO_EDGE
  132     glTexParameteri gl_TEXTURE_2D gl_TEXTURE_WRAP_T
  133         $ fromIntegral gl_CLAMP_TO_EDGE
  134 
  135     depthRenderBuffer <- alloca (\p -> glGenRenderbuffers 1 p >> peek p)
  136 
  137     glBindRenderbuffer gl_RENDERBUFFER depthRenderBuffer
  138 
  139     glRenderbufferStorage gl_RENDERBUFFER gl_DEPTH_COMPONENT
  140         winW winH
  141 
  142     glFramebufferRenderbuffer gl_FRAMEBUFFER gl_DEPTH_ATTACHMENT
  143         gl_RENDERBUFFER depthRenderBuffer
  144 
  145     glFramebufferTexture gl_FRAMEBUFFER
  146        gl_COLOR_ATTACHMENT0 fbTex 0
  147 
  148     withArray [gl_COLOR_ATTACHMENT0] $ glDrawBuffers 1
  149 
  150     glCheckFramebufferStatus gl_FRAMEBUFFER >>=
  151         (\x -> putStrLn $ if x == gl_FRAMEBUFFER_COMPLETE
  152             then "Framebuffer successfully created"
  153             else "Framebuffer error")
  154 
  155     glBindFramebuffer gl_FRAMEBUFFER 0
  156 
  157     quadVB <- fillNewBuffer quadBufferData
  158 
  159     return $ FB fbName fbTex (winW, winH)
  160                 quadVB depthRenderBuffer
  161 
  162 quadBufferData :: [GLfloat]
  163 quadBufferData =
  164     [-1.0, -1.0, 0.0,
  165      1.0, -1.0, 0.0,
  166      -1.0,  1.0, 0.0,
  167      -1.0,  1.0, 0.0,
  168      1.0, -1.0, 0.0,
  169      1.0,  1.0, 0.0]
  170 {-# INLINE quadBufferData #-}
