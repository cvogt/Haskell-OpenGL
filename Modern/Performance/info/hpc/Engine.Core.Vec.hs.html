1 module Engine.Core.Vec (
    2     Vec2(..), Vec3(..), Vec4(..), vec3ToVec4,
    3     normalizeVec3, scaleVec3, lengthVec3,
    4     crossVec3, vec4GetIndex, toArray3, toArray2
    5 ) where
    6 
    7 import Control.DeepSeq (NFData(..))
    8 
    9 import Graphics.Rendering.OpenGL.Raw (GLfloat)
   10 
   11 data Vec4 a = Vec4 a a a a deriving (Show, Eq)
   12 data Vec3 a = Vec3 a a a deriving (Show, Eq)
   13 data Vec2 a = Vec2 a a deriving (Show, Eq)
   14 
   15 instance Functor Vec4 where
   16     fmap f (Vec4 x y z w) = Vec4 (f x) (f y) (f z) (f w)
   17 
   18 instance (Num a) => Num (Vec4 a) where
   19     Vec4 x1 y1 z1 w1 + Vec4 x2 y2 z2 w2 =
   20         Vec4 (x1+x2) (y1+y2) (z1+z2) (w1+w2)
   21     Vec4 x1 y1 z1 w1 - Vec4 x2 y2 z2 w2 =
   22         Vec4 (x1-x2) (y1-y2) (z1-z2) (w1-w2)
   23     Vec4 x1 y1 z1 w1 * Vec4 x2 y2 z2 w2 =
   24         Vec4 (x1*x2) (y1*y2) (z1*z2) (w1*w2)
   25     abs = fmap abs
   26     signum = fmap signum
   27     fromInteger i =
   28         Vec4 (fromInteger i) (fromInteger i) (fromInteger i) (fromInteger i)
   29 
   30 instance Functor Vec3 where
   31     fmap f (Vec3 x y z) = Vec3 (f x) (f y) (f z)
   32 
   33 instance (Num a) => Num (Vec3 a) where
   34     Vec3 x1 y1 z1 + Vec3 x2 y2 z2 = Vec3 (x1+x2) (y1+y2) (z1+z2)
   35     Vec3 x1 y1 z1 - Vec3 x2 y2 z2 = Vec3 (x1-x2) (y1-y2) (z1-z2)
   36     Vec3 x1 y1 z1 * Vec3 x2 y2 z2 = Vec3 (x1*x2) (y1*y2) (z1*z2)
   37     abs (Vec3 x1 y1 z1) = Vec3 (abs x1) (abs y1) (abs z1)
   38     signum = fmap signum
   39     fromInteger i = Vec3 (fromInteger i) (fromInteger i) (fromInteger i)
   40 
   41 instance Functor Vec2 where
   42     fmap f (Vec2 x y) = Vec2 (f x) (f y)
   43 
   44 instance (Num a) => Num (Vec2 a) where
   45     Vec2 x1 y1 + Vec2 x2 y2 = Vec2 (x1+x2) (y1+y2)
   46     Vec2 x1 y1 - Vec2 x2 y2 = Vec2 (x1-x2) (y1-y2)
   47     Vec2 x1 y1 * Vec2 x2 y2 = Vec2 (x1*x2) (y1*y2)
   48     abs (Vec2 x1 y1) = Vec2 (abs x1) (abs y1)
   49     signum = fmap signum
   50     fromInteger i = Vec2 (fromInteger i) (fromInteger i)
   51 
   52 instance NFData (Vec4 a) where
   53     rnf (Vec4 x y z w) = x `seq` y `seq` z `seq` w `seq` ()
   54     {-# INLINE rnf #-}
   55 instance NFData (Vec3 a) where
   56     rnf (Vec3 x y z) = x `seq` y `seq` z `seq` ()
   57     {-# INLINE rnf #-} 
   58 instance NFData (Vec2 a) where
   59     rnf (Vec2 x y) = x `seq` y `seq` ()
   60     {-# INLINE rnf #-}
   61 
   62 vec3ToVec4 :: Vec3 a -> a -> Vec4 a
   63 vec3ToVec4 (Vec3 x y z) = Vec4 x y z
   64 {-# INLINE vec3ToVec4 #-}
   65 
   66 normalizeVec3 :: (Floating a) => Vec3 a -> Vec3 a
   67 normalizeVec3 v =
   68     let Vec3 a b c = scaleVec3 (recip $ lengthVec3 v) v
   69     in Vec3 a b c
   70 {-# SPECIALIZE normalizeVec3 :: Vec3 GLfloat -> Vec3 GLfloat #-}
   71 
   72 scaleVec3 :: (Num a) => a -> Vec3 a -> Vec3 a
   73 scaleVec3 s (Vec3 a b c) = Vec3 (s*a) (s*b) (s*c)
   74 {-# INLINE scaleVec3 #-}
   75 
   76 lengthVec3 :: (Floating a) => Vec3 a -> a
   77 lengthVec3 (Vec3 a b c) = sqrt (a*a + b*b + c*c)
   78 {-# SPECIALIZE lengthVec3 :: Vec3 GLfloat -> GLfloat #-}
   79 
   80 crossVec3 :: (Num a) => Vec3 a -> Vec3 a -> Vec3 a
   81 crossVec3 (Vec3 u0 u1 u2) (Vec3 v0 v1 v2) = 
   82     Vec3 (u1*v2-u2*v1) (u2*v0-u0*v2) (u0*v1-u1*v0)
   83 {-# INLINE crossVec3 #-}
   84 
   85 vec4GetIndex :: Int -> Vec4 a -> a
   86 vec4GetIndex i (Vec4 x y z w)
   87     | i == 0 = x
   88     | i == 1 = y
   89     | i == 2 = z
   90     | i == 3 = w
   91 vec4GetIndex _ _ =
   92     error $ "Vec.vec4GetIndex: Argument must be " ++
   93             "0, 1, 2, or 3."
   94 {-# INLINE vec4GetIndex #-}
   95 
   96 toArray3 :: Vec3 a -> [a]
   97 toArray3 (Vec3 x y z) = [x, y, z]
   98 {-# INLINE toArray3 #-}
   99 
  100 toArray2 :: Vec2 a -> [a]
  101 toArray2 (Vec2 x y) = [x, y]
  102 {-# INLINE toArray2 #-}
