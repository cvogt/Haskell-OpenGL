   34 -- | Create a Framebuffer.
   35 makeShadowFrameBuffer :: (GLint, GLint) -> IO Framebuffer
   36 makeShadowFrameBuffer (width, height) = do
   37     fbName <- alloca (\p -> glGenFramebuffers 1 p >> peek p)
   38     glBindFramebuffer gl_FRAMEBUFFER fbName
   39 
   40     depthTexture <- alloca (\p -> glGenTextures 1 p >> peek p)
   41     glBindTexture gl_TEXTURE_2D depthTexture
   42 
   43     glTexImage2D gl_TEXTURE_2D 0
   44         (fromIntegral gl_DEPTH_COMPONENT16)
   45         width height
   46         0 gl_DEPTH_COMPONENT gl_FLOAT offset0
   47 
   48     -- Give texture paramenters.
   49     glTexParameteri gl_TEXTURE_2D (fromIntegral gl_TEXTURE_MAG_FILTER)
   50                                   (fromIntegral gl_LINEAR)
   51     glTexParameteri gl_TEXTURE_2D (fromIntegral gl_TEXTURE_MIN_FILTER)
   52                                   (fromIntegral gl_LINEAR)
   53     glTexParameteri gl_TEXTURE_2D gl_TEXTURE_WRAP_S
   54         $ fromIntegral gl_CLAMP_TO_EDGE
   55     glTexParameteri gl_TEXTURE_2D gl_TEXTURE_WRAP_T
   56         $ fromIntegral gl_CLAMP_TO_EDGE
   57     glTexParameteri gl_TEXTURE_2D gl_TEXTURE_COMPARE_FUNC
   58         $ fromIntegral gl_LEQUAL
   59     glTexParameteri gl_TEXTURE_2D gl_TEXTURE_COMPARE_MODE
   60         $ fromIntegral gl_COMPARE_R_TO_TEXTURE
   61 
   62     glFramebufferTexture gl_FRAMEBUFFER gl_DEPTH_ATTACHMENT depthTexture 0
   63 
   64     glDrawBuffer gl_NONE
   65 
   66     glCheckFramebufferStatus gl_FRAMEBUFFER >>=
   67         (\x -> putStrLn $ if x == gl_FRAMEBUFFER_COMPLETE
   68             then "Framebuffer successfully created"
   69             else "Framebuffer error")
   70 
   71     return $ FB fbName depthTexture
   72             (width, height)
   73             (-1)
   74             (-1)

   76 -- | Render the world, with shadows!
   77 renderWorldWithShadows :: World t -> IO (World t)
   78 renderWorldWithShadows world = do
   79     let fbuf = fst $ graphicsShadowInfo $ worldGraphics world
   80         depthShader = snd $ graphicsShadowInfo $ worldGraphics world 
   81         (w, h) = fbufDimensions fbuf
   82     glBindFramebuffer gl_FRAMEBUFFER $ fbufName fbuf
   83     glViewport 0 0 w h
   84 
   85     glEnable gl_CULL_FACE
   86     glCullFace gl_FRONT
   87 
   88     glClear $ gl_COLOR_BUFFER_BIT .|. gl_DEPTH_BUFFER_BIT
   89 
   90     glUseProgram depthShader
   91 
   92     let lightInvDir = Vec3 0.5 2.0 2.0
   93         depthProjMat = gorthoMatrix (-50) 50 (-50) 50 (-50) 100
   94         depthViewMat = glookAtMatrix lightInvDir (Vec3 0 0 0) (Vec3 0 1 0)
   95         depthModelMat = gidentityMatrix
   96 
   97         worldMats =
   98             WorldMatrices depthModelMat depthViewMat depthProjMat
   99 
  100     mvpMatrix <- quickGetUniform depthShader "mvpMatrix"
  101 
  102     let depthMVP = depthProjMat * depthViewMat * depthModelMat
  103 
  104     renderInitialShadows (worldEntities world)
  105         worldMats
  106         mvpMatrix
  107 
  108     let win = stateWindow $ worldState world
  109         (width, height) = windowSize win
  110 
  111     glBindFramebuffer gl_FRAMEBUFFER
  112         (fbufName . fst . graphicsPostProcessors . worldGraphics $ world)
  113     glViewport 0 0 (fromIntegral width) (fromIntegral height)
  114 
  115     glEnable gl_CULL_FACE
  116     glCullFace gl_BACK
  117 
  118     glClear $ gl_COLOR_BUFFER_BIT .|. gl_DEPTH_BUFFER_BIT
  119 
  120     renderObjectsWithShadows world
  121         (calculateMatricesFromPlayer (worldPlayer world) (fromIntegral width, fromIntegral height))
  122         depthMVP
  123         fbuf
  124         (worldEntities world)
  125 
  126     let newDepthUniform = ("depthTexture", return [20])
  127         newWorldUniforms =  newDepthUniform :
  128                             graphicsUniforms (worldGraphics world)
  129 
  130         newWorldGraphics = (worldGraphics world){
  131                         graphicsUniforms = newWorldUniforms}
  132     renderAllPasses world{worldGraphics = newWorldGraphics}
  133             (snd $ graphicsPostProcessors $ worldGraphics world)
  134 
  135     return world

  137 renderObjectsWithShadows ::
  138     World t -> WorldMatrices -> Matrix4x4 ->
  139     Framebuffer -> [GameObject t] -> IO ()
  140 renderObjectsWithShadows world wm depthMVP fbuf (cur : rest) = do
  141     let curModel = getModel cur
  142         mShader = modelShader curModel
  143         Vec3 ox oy oz = getPos cur
  144         modelMat = gtranslationMatrix [ox, oy, oz]
  145     glUseProgram $ shaderId mShader
  146 
  147     _ <- setMatrixUniformsBias mShader wm{matrixModel = modelMat} depthMVP
  148 
  149     _ <- setWorldUniforms world mShader
  150 
  151     bindTextures (modelTextures curModel) $ shaderId mShader
  152     
  153     let textureOffset = 20
  154     glActiveTexture $ gl_TEXTURE0 + textureOffset
  155     glBindTexture gl_TEXTURE_2D $ fbufTexture fbuf
  156     quickGetUniform (shaderId mShader) "shadowMap" >>=
  157         (`glUniform1i` fromIntegral textureOffset)
  158 
  159     setShaderAttribs $ modelShaderVars curModel
  160     glDrawArrays gl_TRIANGLES 0 (modelVertCount curModel)
  161 
  162     renderObjectsWithShadows world wm depthMVP fbuf rest
  163 renderObjectsWithShadows _ _ _ _ _ = return ()
  164 
  165 renderInitialShadows :: [GameObject t] -> WorldMatrices -> GLint -> IO ()
  166 renderInitialShadows (cur : rest) wm mvpUniform = do
  167     setShaderAttribs $ modelShaderVars $ getModel cur
  168 
  169     let Vec3 ox oy oz = getPos cur
  170         modelMat = gtranslationMatrix [ox, oy, oz]
  171 
  172         mvp = matrixProjection wm * matrixView wm * modelMat
  173 
  174     withArray
  175         (toGLFormat mvp)
  176         $ glUniformMatrix4fv mvpUniform 1 (fromIntegral gl_FALSE)
  177 
  178     glDrawArrays gl_TRIANGLES 0
  179         (modelVertCount $ getModel cur)
  180     glDisableVertexAttribArray 0
  181     renderInitialShadows rest wm mvpUniform
  182 renderInitialShadows _ _ _ = return ()

  186 setMatrixUniformsBias :: Shader -> WorldMatrices -> Matrix4x4 -> IO Shader
  187 setMatrixUniformsBias shader wm depthMVP = do
  188     (shader', modelMatrix) <- findUniformLocationAndRemember shader "modelMatrix"
  189     withArray (toGLFormat $ matrixModel wm)
  190         $ glUniformMatrix4fv modelMatrix 1 (fromIntegral gl_FALSE)
  191 
  192     (shader'', projectionMatrix) <- findUniformLocationAndRemember shader'
  193                                     "projectionMatrix"
  194     withArray (toGLFormat $ matrixProjection wm)
  195         $ glUniformMatrix4fv projectionMatrix 1 (fromIntegral gl_FALSE)
  196 
  197     (shader''', viewMatrix) <- findUniformLocationAndRemember
  198                                 shader'' "viewMatrix"
  199     withArray (toGLFormat $ matrixView wm)
  200         $ glUniformMatrix4fv viewMatrix 1 (fromIntegral gl_FALSE)
  201 
  202     (shader'''', mvpMatrix) <- findUniformLocationAndRemember
  203                                 shader''' "mvpMatrix"
  204     withArray
  205         (toGLFormat $ matrixProjection wm * matrixView wm * matrixModel wm)
  206         $ glUniformMatrix4fv mvpMatrix 1 (fromIntegral gl_FALSE)
  207 
  208     let biasMatrix = [[0.5, 0.0, 0.0, 0.0],
  209                       [0.0, 0.5, 0.0, 0.0],
  210                       [0.0, 0.0, 0.5, 0.0],
  211                       [0.5, 0.5, 0.5, 1.0]] :: Matrix4x4
  212 
  213     (shader''''', mvpBiasMatrix) <- findUniformLocationAndRemember
  214                                 shader'''' "mvpBiasMatrix"
  215     withArray
  216         (toGLFormat $ biasMatrix * depthMVP)
  217         $ glUniformMatrix4fv mvpBiasMatrix 1 (fromIntegral gl_FALSE)
  218 
  219     return shader'''''
