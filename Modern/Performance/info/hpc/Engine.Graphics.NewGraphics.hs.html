   36     renderInfoShader :: Shader,

   44 -- | A class for things that can be rendered to
   45 --   the screen &| Framebuffers.
   46 --   Contains 3 functions:
   47 --      Bind:
   48 --          This is usually where shaders are set,
   49 --          or the currently bound shader is
   50 --          modified.
   51 --      Draw:
   52 --          This is the draw action. A Framebuffer
   53 --          is bound before calling this, so there is
   54 --          no need to bind your own.
   55 --      Cleanup:
   56 --          Any cleanup necessary.
   57 class Renderable t g where
   58     renderBind :: t -> g -> IO g
   59     renderBind _ = return
   60     renderDraw :: t -> g -> IO g
   61     renderDraw _ = return
   62     renderCleanup :: t -> g -> IO g
   63     renderCleanup _ = return
   64     defaultGlobal :: t -> g
   65 
   66 instance Renderable (GameObject t) RenderInfo where
   67     renderBind obj info =
   68         let shader = modelShader $ getModel obj
   69         in do
   70             glUseProgram $ shaderId shader
   71             return info{renderInfoShader = shader}
   72 
   73     renderDraw object info = do
   74         let model = getModel object
   75             Vec3 objx objy objz = getPos object
   76             Vec3 objrx objry objrz = getRot object
   77             mShader = renderInfoShader info
   78 
   79             -- Move Object
   80             modelMat = gtranslationMatrix [objx, objy, objz] *
   81                        grotationMatrix [objrx, objry, objrz]
   82 
   83             newMatrices = (renderInfoMatrices info){matrixModel = modelMat}
   84         -- Set uniforms. (World uniforms and Matrices).
   85         newShader <-
   86             setMatrixUniforms mShader newMatrices
   87 
   88         -- Bind buffers to variable names in shader.
   89         setShaderAttribs $ modelShaderVars model
   90         bindTextures (modelTextures model) $ shaderId newShader
   91 
   92         glDrawArrays gl_TRIANGLES 0 (modelVertCount model)
   93 
   94         return $ RenderInfo newShader newMatrices
   95 
   96     renderCleanup object _ = do
   97         let model = getModel object
   98         -- Necessary?
   99         disableShaderAttribs $ modelShaderVars model
  100         -- Disable the object's shader.
  101         glUseProgram 0
  102         return emptyInfo
  103     defaultGlobal _ = emptyInfo
  104 
  105 instance Renderable (World t) RenderInfo where
  106     renderBind world info = do
  107         let (winW, winH) = windowSize $ stateWindow $ worldState world
  108             worldMats = calculateMatricesFromPlayer
  109                         (worldPlayer world) (fromIntegral winW, fromIntegral winH)
  110         newShader <-
  111             setWorldUniforms world (renderInfoShader info)
  112         return info{renderInfoShader = newShader,
  113                     renderInfoMatrices = worldMats}
  114     defaultGlobal _ = emptyInfo
  115         
  116 
  117 totalRender :: Renderable t g => t -> g -> IO g
  118 totalRender r s =
  119     renderBind r s >>=
  120     renderDraw r >>=
  121     renderCleanup r
  122 
  123 
  124 renderToFramebuffer :: Renderable t g => Framebuffer -> t -> IO g
  125 renderToFramebuffer fbuf rend =
  126     withFramebuffer fbuf $ totalRender rend (defaultGlobal rend)
  127 
  128 renderAllToFramebuffer :: Renderable t g => Framebuffer -> [t] -> IO g
  129 renderAllToFramebuffer fbuf xs =
  130     withFramebuffer fbuf $
  131         renderAll xs (defaultGlobal $ head xs)
  132     where
  133         renderAll :: Renderable t g => [t] -> g -> IO g
  134         renderAll (x:xs') info =
  135             totalRender x info >>= renderAll xs'
  136         renderAll [] info = return info
  137 
  138 renderAllWithGlobal :: (Renderable t1 g, Renderable t2 g) =>
  139                         Framebuffer -> t1 -> [t2] -> IO g
  140 renderAllWithGlobal fbuf g =
  141     renderAllWithGlobal' (defaultGlobal g) fbuf g
  142 
  143 renderAllWithGlobal' :: (Renderable t1 g, Renderable t2 g) =>
  144                         g -> Framebuffer -> t1 -> [t2] -> IO g
  145 renderAllWithGlobal' info fbuf global (x:xs) = do
  146     bindFramebuffer fbuf
  147 
  148     newinfo <-
  149         renderBind x info >>=
  150         renderBind global >>=
  151         renderDraw global >>=
  152         renderDraw x >>=
  153         renderCleanup x >>=
  154         renderCleanup global
  155 
  156     unbindFrameBuffer
  157 
  158     renderAllWithGlobal' newinfo fbuf global xs
  159 renderAllWithGlobal' info _ _ _ = return info
  160

  162 renderWorldNew :: World t -> IO (World t)
  163 renderWorldNew world = do
  164     glClear $ gl_COLOR_BUFFER_BIT .|. gl_DEPTH_BUFFER_BIT
  165     let (width, height) = windowSize $ stateWindow $ worldState world
  166     _ <- renderAllWithGlobal (screenFramebuffer (fromIntegral width, fromIntegral height))
  167                         world (worldEntities world) :: IO RenderInfo
  168     return world
  169 
  170 renderWorldNewWithFramebuffer :: World t -> Framebuffer -> IO (World t)
  171 renderWorldNewWithFramebuffer world fbuf = do
  172     glClear $ gl_COLOR_BUFFER_BIT .|. gl_DEPTH_BUFFER_BIT
  173     _ <- renderAllWithGlobal fbuf world (worldEntities world) :: IO RenderInfo
  174     return world

  177 -- | Render world with all postprocessing shaders defined by
  178 --   worldFramebuffer.
  179 renderWorldNewPost :: World t -> IO (World t)
  180 renderWorldNewPost world = do
  181     let effects = snd $ graphicsPostProcessors $ worldGraphics world
  182         fb = fst $ graphicsPostProcessors $ worldGraphics world
  183 
  184     bindFramebuffer fb
  185     ret <- renderWorldNewWithFramebuffer world fb
  186 
  187     bindFramebuffer fb
  188     renderAllPasses ret effects
  189     return ret

  192 screenFramebuffer :: (GLint, GLint) -> Framebuffer
  193 screenFramebuffer dimensions =
  194     FB 0 0 dimensions 0 0
  195 
  196 withFramebuffer :: Framebuffer -> IO a -> IO a
  197 withFramebuffer fbuf func = do
  198     bindFramebuffer fbuf
  199     ret <- func
  200     unbindFrameBuffer
  201     return ret
  202 
  203 bindFramebuffer :: Framebuffer -> IO ()
  204 bindFramebuffer = glBindFramebuffer gl_FRAMEBUFFER . fbufName
  205 
  206 unbindFrameBuffer :: IO ()
  207 unbindFrameBuffer = glBindFramebuffer gl_FRAMEBUFFER 0
