   27 instance HasAABB a => Intersect Terrain a where
   28     getIntersecting terrain collider =
   29         let simplex = terrainSimplex terrain
   30             Vec3 x y z = getPos collider
   31             simplexHeight = getSimplexHeight simplex x z
   32         in y <= simplexHeight

   34 generateTerrain :: FilePath -> FilePath ->
   35     GLfloat ->          -- ^ Width
   36     GLfloat ->          -- ^ Spacing
   37     Int ->              -- ^ Octaves
   38     GLfloat ->          -- ^ Wavelength
   39     GLfloat ->          -- ^ Waveheight / intensity
   40     Maybe FilePath ->   -- ^ The texture (Maybe)
   41     IO Terrain
   42 generateTerrain vert frag w spacing octaves wavelength intensity texture = do
   43     seed <- randomRIO (0, 2048)
   44     let simplex = Simplex seed (floor w, floor w) (0, 0) spacing octaves wavelength intensity (perm seed)
   45         vertices = D.toList $ createSimplexTerrain simplex
   46         normals = calculateNormals vertices
   47 
   48     maybe
   49         (loadTerrainWithTexture' simplex vert frag vertices normals undefined)
   50         (loadTerrainWithTexture' simplex vert frag vertices normals)
   51         texture
   52 
   53 loadTerrainWithTexture' :: Simplex -> FilePath -> FilePath -> [GLfloat] -> [GLfloat] -> FilePath -> IO Terrain
   54 loadTerrainWithTexture' simplex vert frag vertices normals texture =
   55     let lengthVertices = length vertices
   56     in do
   57         loadedTerrain <- createTerrain simplex vert frag
   58             ["position", "normal", "color", "texCoord", "textureId"]
   59             [vertices, normals, take (lengthVertices * 3) (cycle [0, 1, 0]),
   60             take (lengthVertices * 3) $ cycle [0, 0, 1, 0, 0, 1],
   61             replicate lengthVertices 0]
   62             [3, 3, 3, 2, 1]
   63             (fromIntegral $ lengthVertices `div` 3)
   64         textureData <- juicyLoadTexture texture
   65         return $ loadedTerrain{terrainTextures = [(textureData, 1)]}
   66 
   67 createTerrain ::
   68     Simplex ->      -- ^ Simplex info.
   69     FilePath ->     -- ^ Vertex Shader.
   70     FilePath ->     -- ^ Fragment Shader.
   71     [String] ->     -- ^ Attribute Variable names.
   72     [[GLfloat]] ->  -- ^ List containing all the lists of values.
   73                    --   (vertices, normals, etc).
   74     [GLuint] ->     -- ^ Size of each value.
   75     GLint ->        -- ^ Number of vertices.
   76     IO Terrain
   77 createTerrain simplex vert frag attrNames buffData valLens vertCount = do
   78     program <- loadProgram vert frag
   79     attribs <- getAttrLocs program attrNames
   80     ids <- createBufferIdAll buffData
   81 
   82     let sAttribs = createShaderAttribs attribs ids valLens
   83     return $ Terrain simplex (Shader program []) sAttribs [] vertCount
   84             $ aabbFromPoints (head buffData)

   86 genSimplexModel :: FilePath -> FilePath ->
   87     GLfloat ->          -- ^ Width
   88     GLfloat ->          -- ^ Spacing
   89     Int ->              -- ^ Octaves
   90     GLfloat ->          -- ^ Wavelength
   91     GLfloat ->          -- ^ Waveheight / intensity
   92     Maybe FilePath ->   -- ^ The texture (Maybe)
   93     IO Model
   94 genSimplexModel vert frag w spacing octaves wavelength intensity texture = do
   95     seed <- randomRIO (0, 2048)
   96     let simplex = Simplex seed (floor w, floor w) (0, 0) spacing octaves wavelength intensity (perm seed)
   97         vertices = D.toList $ createSimplexTerrain simplex
   98         normals = calculateNormals vertices
   99 
  100     maybe
  101         (loadTerrainNoTexture vert frag vertices normals)
  102         (loadTerrainWithTexture vert frag vertices normals)
  103         texture

  105 loadTerrainWithTexture :: FilePath -> FilePath -> [GLfloat] -> [GLfloat] -> FilePath -> IO Model
  106 loadTerrainWithTexture vert frag vertices normals texture =
  107     let lengthVertices = length vertices
  108     in do
  109         loadedModel <- createModel vert frag
  110             ["position", "normal", "color", "texCoord", "textureId"]
  111             [vertices, normals, take (lengthVertices * 3) (cycle [0, 1, 0]),
  112             take (lengthVertices * 3) $ cycle [0, 0, 1, 0, 0, 1],
  113             replicate lengthVertices 0]
  114             [3, 3, 3, 2, 1]
  115             (fromIntegral $ lengthVertices `div` 3)
  116         textureData <- juicyLoadTexture texture
  117         return $ loadedModel{modelTextures = [(textureData, 1)]}
  118 
  119 loadTerrainNoTexture :: FilePath -> FilePath -> [GLfloat] -> [GLfloat] -> IO Model
  120 loadTerrainNoTexture vert frag vertices normals =
  121     let lengthVertices = length vertices
  122     in createModel vert frag
  123             ["position", "normal", "color", "texCoord", "textureId"]
  124             [vertices, normals, take (lengthVertices * 3) (cycle [0, 1, 0]),
  125             replicate (lengthVertices * 3) 0,
  126             replicate lengthVertices (-1)]
  127             [3, 3, 3, 2, 1]
  128             (fromIntegral $ lengthVertices `div` 3)

  130 createSimplexTerrain :: Simplex -> D.DList GLfloat
  131 createSimplexTerrain simplex =
  132     concatMapD (\x -> concatMapD (makeSquare simplex x) $ allYs simplex) $ allXs simplex
  133 
  134 concatMapD :: (a -> D.DList b) -> D.DList a -> D.DList b
  135 concatMapD f = D.foldr (D.append . f) D.empty
  136 
  137 allXs :: Simplex -> D.DList GLfloat
  138 allXs simplex =
  139     let first = fromIntegral $ fst $ simpStartXY simplex :: GLfloat
  140     in first `D.cons` allXs' simplex first 0
  141     where
  142         allXs' :: Simplex -> GLfloat -> Int -> D.DList GLfloat
  143         allXs' s i j
  144             | j < fst (simpDimensions s) =
  145                 let cur = i + simpSpacing s
  146                 in cur `D.cons` allXs' simplex cur (j+1)
  147             | otherwise = D.empty
  148 
  149 allYs :: Simplex -> D.DList GLfloat
  150 allYs simplex =
  151     let first = fromIntegral $ snd $ simpStartXY simplex :: GLfloat
  152     in first `D.cons` allYs' simplex first 0
  153     where
  154         allYs' :: Simplex -> GLfloat -> Int -> D.DList GLfloat
  155         allYs' s i j
  156             | j < snd (simpDimensions s) =
  157                 let cur = i + simpSpacing s
  158                 in cur `D.cons` allYs' simplex cur (j+1)
  159             | otherwise = D.empty
  160 
  161 makeSquare :: Simplex -> GLfloat -> GLfloat -> D.DList GLfloat
  162 makeSquare simplex x z =
  163     let spacing = simpSpacing simplex
  164     in makePointFromXY simplex x z `D.append`
  165         makePointFromXY simplex x (z+spacing) `D.append`
  166         makePointFromXY simplex (x+spacing) z `D.append`
  167         makePointFromXY simplex (x+spacing) z `D.append`
  168         makePointFromXY simplex x (z+spacing) `D.append`
  169         makePointFromXY simplex (x+spacing) (z+spacing)
  170 
  171 makePointFromXY :: Simplex -> GLfloat -> GLfloat -> D.DList GLfloat
  172 makePointFromXY simp x z = D.fromList [x, getSimplexHeight simp x z, z]

  208 applyHeights :: [GLfloat] -> [GLfloat] -> [GLfloat]
  209 applyHeights (x:y:z:coords) ref =
  210     let foundY = findY x z ref
  211     in x : (if foundY == -3.1415
  212         then y
  213     else foundY) : z : applyHeights coords ref
  214 applyHeights _ _ = []

  216 findY :: GLfloat -> GLfloat -> [GLfloat] -> GLfloat
  217 findY xf zf (xl:yl:zl:heights)
  218     | xf == xl && zf == zl = yl
  219     | otherwise = findY xf zf heights
  220 findY _ _ _ = -3.1415

  223 calculateNormals :: [GLfloat] -> [GLfloat]
  224 calculateNormals (x1:y1:z1:x2:y2:z2:x3:y3:z3:rest) =
  225     let (ux, uy, uz) = (x2 - x1, y2 - y1, z2 - z1)
  226         (vx, vy, vz) = (x3 - x1, y3 - y1, z3 - z1)
  227         nx = (uy * vz) - (uz * vy)
  228         ny = (uz * vx) - (ux * vz)
  229         nz = (ux * vy) - (uy * vx)
  230         -- Repeat this normal for 6 points (3 points for this half
  231         -- of the square, 3 points for the other half). Drop the next
  232         -- 3 points (9 floats) - the next triangle.
  233     in repeatList 6 [nx, if ny > 0 then ny else -ny, nz] ++
  234             calculateNormals (drop 9 rest)
  235     where
  236         repeatList :: Int -> [a] -> [a]
  237         repeatList i list = take (i*3) $ cycle list 
  238 calculateNormals _ = []
