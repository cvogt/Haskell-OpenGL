   33 fastFloor :: Double -> Int
   34 fastFloor x = truncate $ if x > 0 then x else x - 1

   36 -- | Generate a random permutation for use in the noise functions
   37 perm :: Int -> Permutation
   38 perm seed = V.fromList . concat . replicate 2 . shuffle' [0..255] 256 $
   39                 mkStdGen seed

   41 -- | Generate 3D noise between -0.5 and 0.5
   42 noise3D :: Permutation -> Double -> Double -> Double -> Double
   43 noise3D p x y z =
   44     16 * (n gi0 (x0,y0,z0) + n gi1 xyz1 + n gi2 xyz2 + n gi3 xyz3)
   45     where
   46     (i,j,k) = (s x, s y, s z)
   47     s a = fastFloor (a + (x + y + z) / 3)
   48     (x0, y0, z0) = (x - int i + t, y - int j + t, z - int k + t)
   49     t = int (i + j + k) * g3
   50     (i1, j1, k1, i2, j2, k2)
   51         | x0 >= y0 =
   52             if y0 >= z0
   53                 then (1,0,0,1,1,0)
   54             else if x0 >= z0
   55                 then (1,0,0,1,0,1)
   56             else (0,0,1,1,0,1)
   57         | x0 >= y0 =
   58                 if y0 >= z0
   59                     then (1,0,0,1,1,0)
   60                 else if x0 >= z0
   61                     then (1,0,0,1,0,1)
   62                 else (0,0,1,1,0,1)
   63         | y0 <  z0 = (0,0,1,0,1,1)
   64         | x0 <  z0 = (0,1,0,0,1,1)
   65         | otherwise = (0,1,0,1,1,0)
   66     xyz1 = (x0 - int i1 +   g3, y0 - int j1 +   g3, z0 - int k1 +   g3)
   67     xyz2 = (x0 - int i2 + 2*g3, y0 - int j2 + 2*g3, z0 - int k2 + 2*g3)
   68     xyz3 = (x0 - 1 + 3*g3, y0 - 1 + 3*g3, z0 - 1 + 3*g3)
   69     (ii,jj,kk) = (i .&. 255, j .&. 255, k .&. 255)
   70     gi0 = rem (V.unsafeIndex p 
   71         (ii + V.unsafeIndex p (jj + V.unsafeIndex p kk))) 12
   72     gi1 = rem (V.unsafeIndex p 
   73         (ii + i1 + V.unsafeIndex p (jj + j1 + V.unsafeIndex p (kk + k1)))) 12
   74     gi2 = rem (V.unsafeIndex p
   75         (ii + i2 + V.unsafeIndex p (jj + j2 + V.unsafeIndex p (kk + k2)))) 12
   76     gi3 = rem (V.unsafeIndex p
   77         (ii + 1  + V.unsafeIndex p (jj + 1  + V.unsafeIndex p (kk + 1)))) 12
   78     {-# INLINE n #-}
   79     n gi (x',y',z') = (\a -> if a < 0 then 0 else
   80         a*a*a*a*dot3 (V.unsafeIndex grad3 gi) x' y' z') $
   81             0.6 - x'*x' - y'*y' - z'*z'

   83 harmonic :: Int -> (Double -> Double) -> Double
   84 harmonic octaves noise = f octaves / (2 - 1 / int (2 ^ (octaves - 1)))
   85     where
   86     f 0 = 0
   87     f o = let r = 2 ^^ (o - 1) in noise r / r + f (o - 1)
   88 
   89 -- | 3D simplex noise
   90 --   args - permutation number_of_octaves wavelength x y z
   91 simplex3D :: Permutation -> Int -> Double -> Double -> Double -> Double -> Double
   92 simplex3D p octaves l x y z = harmonic octaves
   93     (\f -> noise3D p (x * f / l) (y * f / l) (z * f / l))
   94 
   95 getSimplexHeight :: Simplex -> GLfloat -> GLfloat -> GLfloat
   96 getSimplexHeight (Simplex _ _ _ _ octaves wavelength intensity permutation) x y =
   97         intensity * realToFrac
   98             (simplex3D permutation octaves (realToFrac wavelength)
   99                 (realToFrac x) (realToFrac y) 0)
  100 
  101 -- | Generate a 2D list of the height returned by simplex
  102 --   for each coordinate.
  103 simplexNoise :: Int -> GLfloat -> Int -> GLfloat -> GLfloat -> IO [[GLfloat]]
  104 simplexNoise width spacing octaves wavelength intensity = do
  105     seed <- randomRIO (0, 100)
  106     let raw = map
  107             (\(x, y) -> intensity * realToFrac
  108                 (simplex3D (perm seed) octaves (realToFrac wavelength) x y 0))
  109             [(realToFrac x, realToFrac y) |
  110                 x <- [0, spacing .. (fromIntegral $ width-1)],
  111                 y <- [0, spacing .. (fromIntegral $ width-1)]]
  112     return $ splitInterval raw width

  127 splitInterval :: [a] -> Int -> [[a]]
  128 splitInterval xs i
  129     | not $ null xs =
  130         let (ret, rest) = splitAt i xs
  131         in ret : splitInterval rest i
  132     | otherwise = []
