   29 output :: String -> a -> a
   30 output msg = seq (unsafePerformIO $ putStrLn msg)

   33 findNearby :: HasAABB a => Octree a -> a -> [a]
   34 findNearby (ONode _ children) val =
   35     let insertIntos = filter (checkOctant val) children
   36     in if null insertIntos
   37             then []
   38         else concatMap (`findNearby` val) insertIntos
   39 findNearby (OLeaf _ contents _) _ = contents
   40 
   41 octInsert :: (HasAABB a, Show a) => Octree a -> a -> Octree a
   42 octInsert tree@(ONode aabb children) val =
   43     let (insertIntos, others) = filterPartition (checkOctant val) children
   44     in if null insertIntos
   45             then output
   46                 ("Error in Collision.octInsert:" ++
   47                  " value \"" ++ show val ++
   48                  "\"could not be inserted into tree. Ignoring.") tree
   49         else ONode aabb $ map (`octInsert` val) insertIntos ++ others
   50 octInsert leaf@(OLeaf aabb contents size) val =
   51     if size+1 <= maxCapacity
   52         then OLeaf aabb (val : contents) (size+1)
   53     else octInsert (subdivide leaf) val
   54 
   55 subdivide :: (HasAABB a, Show a) => Octree a -> Octree a
   56 subdivide (OLeaf wholeAABB@(AABB minVec maxVec) contents _) =
   57     let halfVec@(Vec3 halfX halfY halfZ) = fmap (/2) (abs $ maxVec - minVec)
   58         newAABBTemplate = AABB minVec $ minVec + halfVec 
   59 
   60         northWestA = newAABBTemplate
   61         northWestB = movePos northWestA (Vec3 0 0 halfZ)
   62         northWestALeaf = OLeaf northWestA [] 0
   63         northWestBLeaf = OLeaf northWestB [] 0
   64         
   65         northEastA = movePos newAABBTemplate (Vec3 halfX 0 0)
   66         northEastB = movePos northEastA (Vec3 0 0 halfZ)
   67         northEastALeaf = OLeaf northEastA [] 0
   68         northEastBLeaf = OLeaf northEastB [] 0
   69 
   70         southWestA = movePos newAABBTemplate (Vec3 0 halfY 0)
   71         southWestB = movePos southWestA (Vec3 0 0 halfZ)
   72         southWestALeaf = OLeaf southWestA [] 0
   73         southWestBLeaf = OLeaf southWestB [] 0
   74 
   75         southEastA = movePos newAABBTemplate (Vec3 halfX halfY 0)
   76         southEastB = movePos southEastA (Vec3 0 0 halfZ)
   77         southEastALeaf = OLeaf southEastA [] 0
   78         southEastBLeaf = OLeaf southEastB [] 0
   79  
   80         newNode = ONode wholeAABB [northWestALeaf, northWestBLeaf,
   81                                    northEastALeaf, northEastBLeaf,
   82                                    southWestALeaf, southWestBLeaf,
   83                                    southEastALeaf, southEastBLeaf]
   84     in foldl octInsert newNode contents
   85 subdivide _ = error "Collision.subdivide: cannot subdivide a ONode."
   86 
   87 filterPartition :: (a -> Bool) -> [a] -> ([a], [a])
   88 filterPartition f (x:xs)
   89     | f x = consTuple (Just x, Nothing) $ filterPartition f xs
   90     | otherwise = consTuple (Nothing, Just x) $ filterPartition f xs
   91 filterPartition _ [] = ([], [])
   92 
   93 consTuple :: (Maybe a, Maybe a) -> ([a], [a]) -> ([a], [a])
   94 consTuple (Just x, Nothing) (xs, ys) = (x:xs, ys)
   95 consTuple (Nothing, Just y) (xs, ys) = (xs, y:ys)
   96 consTuple _ _ =
   97     error "Octree.consTuple: invalid arguments."
   98 
   99 checkOctant :: HasAABB a => a -> Octree a -> Bool
  100 checkOctant val (ONode aabb _) =
  101     objectsIntersect val aabb
  102 checkOctant val (OLeaf aabb _ _) =
  103     objectsIntersect val aabb
  104 {-# INLINE checkOctant #-}
